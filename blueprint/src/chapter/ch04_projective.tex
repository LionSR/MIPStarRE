\begin{theorem}[Naimark dilation]\label{thm:naimark}
	\lean{MIPStarRE.LDT.MakingMeasurementsProjective.naimark}
	\leanok
	\uses{def:submeasurement}
	Let $\ket{\psi}$ be a state in $\mathcal H_{\mathrm A} \ot \mathcal H_{\mathrm B}$.
	Let $A = \{A^x_a\}$ be a sub-measurement acting on $\mathcal H_{\mathrm A}$
	and $B=\{B^y_b\}$ be a sub-measurement acting on $\mathcal H_{\mathrm B}$.
	Then there exist Hilbert spaces $\mathcal H_{\mathrm A_{\mathsf{aux}}}$ and $\mathcal H_{\mathrm B_{\mathsf{aux}}}$,
	a state $\ket{\mathsf{aux}} \in \mathcal H_{\mathrm A_{\mathsf{aux}}} \ot \mathcal H_{\mathrm B_{\mathsf{aux}}}$,
	and measurements $\widehat{A} = \{\widehat{A}^x_a\}$ and $\widehat{B}=\{\widehat{B}^y_b\}$
	acting on $\mathcal H_{\mathrm A} \ot \mathcal H_{\mathrm A_{\mathsf{aux}}}$
	and $\mathcal H_{\mathrm B} \ot \mathcal H_{\mathrm B_{\mathsf{aux}}}$, respectively,
	such that, if $\ket{\widehat{\psi}} = \ket{\psi} \ot \ket{\mathsf{aux}}$, then for all $x,y,a,b$,
	\[
		\bra{\psi} A^x_a \ot B^y_b \ket{\psi}
		=
		\bra{\widehat{\psi}} \widehat{A}^x_a \ot \widehat{B}^y_b \ket{\widehat{\psi}}.
	\]
	In addition, $\ket{\mathsf{aux}}$ is a product state.
\end{theorem}
\begin{proof}
	\uses{lem:naimark-helper}

	For each question $x$, choose a local auxiliary state
	$\ket{\mathsf{aux}_{\mathrm A,x}}$ of dimension one larger than the number of
	outcomes of $A^x$, and apply Lemma~\ref{lem:naimark-helper} to obtain a
	projective sub-measurement $\widetilde A^x$. Do the same for each question $y$
	on Bob's side, obtaining auxiliary states $\ket{\mathsf{aux}_{\mathrm B,y}}$
	and projective sub-measurements $\widetilde B^y$.

	Let
	\[
		\ket{\mathsf{aux}}
		=
		\Big(\bigotimes_x \ket{\mathsf{aux}_{\mathrm A,x}}\Big)
		\ot
		\Big(\bigotimes_y \ket{\mathsf{aux}_{\mathrm B,y}}\Big),
		\qquad
		\ket{\widehat{\psi}} = \ket{\psi} \ot \ket{\mathsf{aux}},
	\]
	and let $\widehat A^x_a$ and $\widehat B^y_b$ be obtained by letting
	$\widetilde A^x_a$ or $\widetilde B^y_b$ act on the relevant auxiliary factor
	and the identity act on every other auxiliary factor. Because the auxiliary
	state is a product state and the two dilations act on disjoint tensor factors,
	the correlation for fixed $(x,y)$ reduces to the compression identity from
	Lemma~\ref{lem:naimark-helper} on the $x$- and $y$-registers. Hence
	\[
		\bra{\widehat{\psi}} \widehat A^x_a \ot \widehat B^y_b \ket{\widehat{\psi}}
		=
		\bra{\psi} A^x_a \ot B^y_b \ket{\psi}.
	\]
\end{proof}

\begin{theorem}[Orthogonalization lemma]\label{thm:orthonormalization}
	\lean{MIPStarRE.LDT.MakingMeasurementsProjective.orthonormalization}
	\leanok
	\uses{def:strong-self-consistency, def:approx_delta}
	Let $\ket{\psi}$ be a permutation-invariant state,
	and let $A = \{A_a\}$ be a sub-measurement with strong self-consistency
	\[
		\sum_a \bra{\psi} A_a \ot A_a \ket{\psi}
		\geq
		\sum_a \bra{\psi} A_a \ot I \ket{\psi} - \zeta.
	\]
	Then there exists a projective sub-measurement $P = \{P_a\}$ such that
	\[
		A_a \ot I \approx_{100 \zeta^{1/4}} P_a \ot I.
	\]
\end{theorem}

\subsection{Naimark dilation}\label{sec:naimark}

\begin{lemma}\label{lem:naimark-helper}
	\lean{MIPStarRE.LDT.MakingMeasurementsProjective.oneMeasNaimark}
	\leanok
	Let $A = \{A_a\}$ be a sub-measurement with $k$ distinct outcomes $a \in \mathcal A$, and let $\ket{\mathsf{aux}} \in \C^{k+1}$ be any state. Then there exists a projective sub-measurement $\widehat{A} = \{\widehat{A}_a\}$ such that for each outcome $a$,
	\[
		(I \ot \bra{\mathsf{aux}}) \cdot \widehat{A}_a \cdot (I \ot \ket{\mathsf{aux}})
		=
		A_a.
	\]
\end{lemma}
\begin{proof}
	Consider an orthonormal basis for $\C^{k+1}$ consisting of vectors $\ket{a}$ for $a \in \mathcal A$ and the vector $\ket{\bot}$. Let $U$ be any unitary such that for each vector $\ket{\phi}$,
	\[
		U \cdot (\ket{\phi} \ot \ket{\mathsf{aux}})
		=
		\sum_{a \in \mathcal A} ((A_a)^{1/2}\ket{\phi}) \ot \ket{a}
		+
		((I-A)^{1/2}\ket{\phi}) \ot \ket{\bot},
	\]
	where $A=\sum_a A_a$. Then
	\[
		U \cdot (I \ot \ket{\mathsf{aux}})
		=
		\sum_{a \in \mathcal A} (A_a)^{1/2} \ot \ket{a}
		+
		(I-A)^{1/2} \ot \ket{\bot},
	\]
	so for any $a \in \mathcal A$,
	\[
		(I \ot \bra{a}) \cdot U \cdot (I \ot \ket{\mathsf{aux}})
		=
		(A_a)^{1/2}.
	\]
	Define
	\[
		\widehat{A}_a = U^\dagger \cdot (I \ot \ket{a}\bra{a}) \cdot U.
	\]
	Each $\widehat{A}_a$ is a projector, and
	\begin{align*}
		(I \ot \bra{\mathsf{aux}}) \cdot \widehat{A}_a \cdot (I \ot \ket{\mathsf{aux}})
		 & =
		(I \ot \bra{\mathsf{aux}}) \cdot U^\dagger \cdot (I \ot \ket{a}) \cdot (I \ot \bra{a}) \cdot U \cdot (I \ot \ket{\mathsf{aux}}) \\
		 & =
		(A_a)^{1/2} \cdot (A_a)^{1/2}
		=
		A_a.
	\end{align*}
\end{proof}

\begin{remark}[Naimark dilation does not preserve $\approx_\delta$]\label{ex:easy-but-long}
		Let $\ket{\psi}$ be any state in $\C^d \ot \C^d$, and consider the two-outcome measurements $A = \{A_0, A_1\}$ and $B = \{B_0, B_1\}$ with
		\[
			A_0 = A_1 = B_0 = B_1 = \tfrac{1}{2} I.
		\]
		For each outcome $a \in \{0,1\}$, the operators $A_a \ot I$ and $I \ot B_a$ are both equal to $\tfrac{1}{2} I \ot I$, so $A_a \ot I \approx_0 I \ot B_a$ on the state $\ket{\psi}$.

		Now apply the Naimark construction with a qubit auxiliary space on each side and auxiliary state $\ket{+} = \tfrac{1}{\sqrt 2}(\ket{0}+\ket{1})$. Since $A$ and $B$ are genuine measurements rather than sub-measurements, the extra $\ket{\bot}$ direction from \ref{lem:naimark-helper} is unnecessary in this example. The unitary from \ref{lem:naimark-helper} can then be taken to be the identity, so the dilated measurements are
		\[
			\widehat A_a = I \ot \ket{a}\bra{a}, \qquad \widehat B_a = I \ot \ket{a}\bra{a}.
		\]
		If $\ket{\widehat{\psi}} = \ket{\psi} \ot \ket{+} \ot \ket{+}$, then a direct calculation gives
		\[
			\sum_{a \in \{0,1\}} \bigl\lVert (\widehat A_a \ot I - I \ot \widehat B_a) \ket{\widehat{\psi}} \bigr\rVert^2 = 1.
		\]
		Thus on the state $\ket{\widehat{\psi}}$ the dilated family satisfies $\widehat A_a \ot I \approx_\delta I \ot \widehat B_a$ only for $\delta \ge 1$, even though the original family satisfied the same relation with $\delta = 0$. This is why we use Theorem~\ref{thm:orthonormalization} rather than simply applying Theorem~\ref{thm:naimark}: Naimark preserves the outcome probabilities, but it need not preserve the state-dependent $\approx_\delta$ estimates needed later.
\end{remark}

\subsection{Orthogonalization lemma}\label{sec:orthogonalization}

\begin{lemma}[Orthogonalization lemma for measurements]\label{lem:orthonormalization-main-lemma}
	\lean{MIPStarRE.LDT.MakingMeasurementsProjective.orthonormalizationMainLemma}
	\leanok
	\uses{def:simeq, def:approx_delta}
	Let $\ket{\psi}$ be a state, and let $A = \{A_a\}$ and $B = \{B_a\}$ be measurements such that
	\[
		A_a \ot I \simeq_\zeta I \ot B_a
	\]
	on state $\ket{\psi}$. Here $\ket{\psi}$ is not assumed to be permutation-invariant.
	Then there exists a projective sub-measurement $P = \{P_a\}$ such that
	\[
		A_a \ot I \approx_{84 \zeta^{1/4}} P_a \ot I.
	\]
\end{lemma}

\begin{lemma}[Local orthogonalization lemma for a measurement]\label{lem:orthonormalization-main-lemma-local}
	\lean{MIPStarRE.LDT.MakingMeasurementsProjective.orthonormalizationMainLemma_local}
	\leanok
	\uses{def:strong-self-consistency, def:approx_delta}
	Let $\ket{\psi}$ be a unit vector, let $A = \{A_a\}$ be a measurement, and
	let $0 \leq \zeta \leq 1$. Assume bipartite strong self-consistency
	\[
		\sum_a \bra{\psi} A_a \ot A_a \ket{\psi}
		\geq
		\sum_a \bra{\psi} A_a \ot I \ket{\psi} - \zeta,
	\]
	together with a spectral-truncation witness for the left-lifted family
	$\{A_a \ot I\}$ and a locality-preserving repair witness producing
	projective families of the form $\widehat P_a = P_a \ot I$. Then there
	exists a projective sub-measurement $P = \{P_a\}$ on the same space as
	$A$ such that
	\[
		A_a \ot I \approx_{84 \zeta^{1/4}} P_a \ot I.
	\]
\end{lemma}
\begin{proof}
	\leanok
	\uses{lem:orthonormalization-main-lemma}
	Since $A$ is a measurement, $\sum_a A_a = I$, so bipartite strong
	self-consistency of $A$ with itself is equivalent to bipartite consistency
	of $A$ with itself. Feed this consistency hypothesis, together with the
	spectral-truncation and locality-preserving repair witnesses, into the
	measurement variant of \ref{lem:orthonormalization-main-lemma} with
	$B = A$. The repair witness chooses the repaired family of the form
	$P_a \ot I$, so the conclusion descends to a local projective
	sub-measurement $P = \{P_a\}$ on the space of $A$ with error bound
	$84\zeta^{1/4}$.
\end{proof}

\begin{theorem}[Orthogonalization for a measurement]\label{thm:orthonormalization-measurement}
	\lean{MIPStarRE.LDT.MakingMeasurementsProjective.orthonormalizationMeasurement}
	\leanok
	\uses{def:strong-self-consistency, def:approx_delta}
	Let $\ket{\psi}$ be a unit vector, let $A = \{A_a\}$ be a measurement, and
	let $0 \leq \zeta \leq 1$. Assume bipartite strong self-consistency of $A$
	together with the spectral-truncation and locality-preserving repair
	witnesses for the left-lifted family $\{A_a \ot I\}$. Then there exists a
	projective sub-measurement $P = \{P_a\}$ such that
	\[
		A_a \ot I \approx_{100 \zeta^{1/4}} P_a \ot I.
	\]
\end{theorem}
\begin{proof}
	\leanok
	\uses{lem:orthonormalization-main-lemma-local}
	Apply \ref{lem:orthonormalization-main-lemma-local} and weaken the error
	using $84 \leq 100$ and $\zeta^{1/4} \geq 0$. No completion step is
	needed because $A$ is already a measurement.
\end{proof}

\begin{proof}[Proof of \ref{thm:orthonormalization} assuming \ref{lem:orthonormalization-main-lemma}]
	\proves{thm:orthonormalization}
	\leanok
	\uses{prop:other-two-notions-of-self-consistency, lem:orthonormalization-main-lemma}

	Let $A = \{A_a\}$ be a sub-measurement with outcomes in $\mathcal A$. Write $A=\sum_a A_a$. The strong self-consistency assumption implies
	\[
		\bra{\psi} A \ot A \ket{\psi}
		\geq
		\bra{\psi} A \ot I \ket{\psi} - \zeta,
	\]
	so
	\[
		\bra{\psi} A \ot (I-A) \ket{\psi} \leq \zeta.
	\]
	Rearranging once more gives
	\[
		\bra{\psi} (I-A) \ot (I-A) \ket{\psi}
		\geq
		\bra{\psi} (I-A) \ot I \ket{\psi} - \zeta.
	\]

	Complete $A$ to the measurement $\widehat A=\{\widehat A_a\}_{a \in \widehat{\mathcal A}}$ on $\widehat{\mathcal A}=\mathcal A \cup \{\bot\}$ by setting $\widehat A_a=A_a$ for $a \in \mathcal A$ and $\widehat A_{\bot}=I-A$. Adding the two preceding inequalities shows that $\widehat A$ is $2\zeta$-strongly self-consistent. By \ref{prop:other-two-notions-of-self-consistency}, this is equivalent to
	\[
		\widehat A_a \ot I \simeq_{2\zeta} I \ot \widehat A_a.
	\]
	Apply \ref{lem:orthonormalization-main-lemma} to obtain a projective sub-measurement $\widehat P$ such that
	\[
		\widehat A_a \ot I \approx_{84(2\zeta)^{1/4}} \widehat P_a \ot I.
	\]
	Discard the extra outcome and set $P_a=\widehat P_a$ for $a \in \mathcal A$. Then
	\[
		A_a \ot I \approx_{84(2\zeta)^{1/4}} P_a \ot I,
	\]
	and $84 \cdot 2^{1/4} \leq 100$, so the claimed bound follows.
\end{proof}

\begin{proof}[Proof of \ref{lem:orthonormalization-main-lemma}]
	\proves{lem:orthonormalization-main-lemma}
	\uses{prop:simeq-for-measurements, lem:projective-non-measurement, lem:projective-low-rank-sum, lem:Q-completeness, lem:sqrt-Q-completeness, lem:q-almost-projective, lem:xa-t, lem:qa-restated, lem:X-squared, lem:X-expression-to-Q-expression, lem:pa-restated, lem:X-hat-squared, lem:X-times-X-hat, lem:squared-difference, lem:P-projectivity, lem:P-Q-approx, prop:triangle-inequality-for-approx_delta}
	The statement is trivial when $\zeta > 1/4$, so assume
	\begin{equation}\label{eq:assumption-on-zeta}
		\zeta \leq 1/4.
	\end{equation}
	By \ref{prop:simeq-for-measurements},
	\[
		\sum_a \bra{\psi} A_a \ot B_a \ket{\psi} \geq 1-\zeta.
	\]
	Apply Cauchy--Schwarz to
	\[
		\sum_a \bra{\psi} (A_a \ot I)\cdot(I \ot B_a)\ket{\psi}
	\]
	to obtain
	\[
		\sum_a \bra{\psi} (A_a)^2 \ot I \ket{\psi} \geq (1-\zeta)^2 \geq 1-2\zeta.
	\]
	Because $A$ is a measurement, this rewrites as
	\begin{equation}\label{eq:A-looks-projective}
		\sum_a \bra{\psi} (A_a-(A_a)^2) \ot I \ket{\psi} \leq 2\zeta.
	\end{equation}

	\begin{lemma}\label{lem:trunc-inequality}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.truncationInequality}
		\leanok
		For any $x \in [0, 1]$,
		\begin{equation*}
			(x - \mathsf{trunc}_\delta(x))^2 \leq \frac{1}{\delta} \cdot (x - x^2).
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\leanok
		If $x \geq 1-\delta$, then $\mathsf{trunc}_\delta(x)=1$. Since $\delta \leq 1/2$ below, one also has $x \geq \delta$, so
		\[
			(x-\mathsf{trunc}_\delta(x))^2
			=
			(1-x)^2
			\leq
			(1-x)\frac{x}{\delta}
			=
			\frac{1}{\delta}(x-x^2).
		\]
		If $x < 1-\delta$, then $\mathsf{trunc}_\delta(x)=0$, and similarly
		\[
			(x-\mathsf{trunc}_\delta(x))^2
			=
			x^2
			\leq
			x\frac{1-x}{\delta}
			=
			\frac{1}{\delta}(x-x^2).
		\]
	\end{proof}

	\begin{lemma}[Rounding to projectors]\label{lem:projective-non-measurement}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.projectiveNonMeasurement}
		\leanok
		There exists a set of projective matrices $\{R_a\}$ such that
		\begin{equation*}
			A_a \ot I \approx_{2\sqrt{\zeta}} R_a \ot I.
		\end{equation*}
		and
		\begin{equation*}
			R:= \sum_a R_a \leq (1+2\sqrt{\zeta}) \cdot I.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\uses{lem:trunc-inequality}
		Write the eigendecomposition of $A_a$ as
		\[
			A_a = \sum_i \lambda_{a,i}\ket{u_{a,i}}\bra{u_{a,i}}.
		\]
		Choose
		\begin{equation}\label{eq:bound-on-delta}
			0 < \delta \leq 1/2,
		\end{equation}
		define $\mathsf{trunc}_\delta:[0,1] \to \{0,1\}$ by $\mathsf{trunc}_\delta(x)=1$ when $x \geq 1-\delta$ and $0$ otherwise, and set
		\[
			R_a = \mathsf{trunc}_\delta(A_a)
			=
			\sum_i \mathsf{trunc}_\delta(\lambda_{a,i}) \ket{u_{a,i}}\bra{u_{a,i}}.
		\]
		By \ref{lem:trunc-inequality},
		\[
			(A_a-R_a)^2
			\leq
			\frac{1}{\delta}(A_a-(A_a)^2).
		\]
		Summing expectations and using \ref{eq:A-looks-projective} gives
		\[
			\sum_a \Vert (A_a-R_a)\ot I \ket{\psi}\Vert^2
			\leq
			\frac{2\zeta}{\delta},
		\]
		so $A_a \ot I \approx_{2\zeta/\delta} R_a \ot I$.

		For every $x \in [0,1]$, one has $\mathsf{trunc}_\delta(x) \leq x/(1-\delta)$, hence
		\[
			R_a \leq \frac{1}{1-\delta} A_a.
		\]
		Summing over $a$ and using that $A$ is a measurement,
		\[
			R = \sum_a R_a \leq \frac{1}{1-\delta} I \leq (1+2\delta)I,
		\]
		where the last estimate uses $\delta \leq 1/2$. Setting $\delta=\sqrt{\zeta}$ gives the claim, and \ref{eq:assumption-on-zeta} ensures $\delta \leq 1/2$.
	\end{proof}

	\begin{lemma}[Combinatorial averaging for the $r > d$ branch]\label{lem:small-overlap-averaging}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.Truncation.card_mul_sum_small_le,
			MIPStarRE.LDT.MakingMeasurementsProjective.Truncation.card_univ_mul_sum_compl_le,
			MIPStarRE.LDT.MakingMeasurementsProjective.Truncation.sum_small_le_four_sqrt}
		\leanok
		Let $f : \alpha \to \mathbb{R}$ be a function on a finite index set $\alpha$ with
		$|\alpha| = r$, and let $L \subseteq \alpha$ with $|L| = d$ and $d < r$. Write
		$S = L^\complement$ and assume $f(s) \leq f(l)$ for every $s \in S$ and $l \in L$.
		Then the pairwise double-counting inequality
		\[
			|L|\, \sum_{s \in S} f(s) \leq |S|\, \sum_{l \in L} f(l)
		\]
		holds, and hence also
		\[
			r\, \sum_{s \in S} f(s) \leq |S|\, \sum_{x \in \alpha} f(x).
		\]
		If in addition $r \leq (1+2\sqrt{\zeta})\,d$, $\sum_{x \in \alpha} f(x) \leq
		1 + 2\sqrt{\zeta}$, and $0 \leq \zeta \leq 1/4$, then
		\[
			\sum_{s \in S} f(s) \leq 4\sqrt{\zeta}.
		\]
	\end{lemma}

	\begin{lemma}[Rank reduction]\label{lem:projective-low-rank-sum}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.projectiveLowRankSum}
		\leanok
		There exists a set of projection matrices $\{Q_a\}$ such that
		\begin{equation*}
			A_a \ot I \approx_{12\sqrt{\zeta}} Q_a \ot I.
		\end{equation*}
		and
		\begin{equation*}
			Q:= \sum_a Q_a \leq (1+2\sqrt{\zeta}) \cdot I.
		\end{equation*}
		Furthermore,
		\begin{equation*}
			\sum_a \mathrm{rank}(Q_a) \leq d.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\uses{lem:projective-non-measurement, lem:small-overlap-averaging, prop:triangle-inequality-for-approx_delta}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.projectiveLowRankSum_of_projectors}
		Let $\{R_a\}$ be given by \ref{lem:projective-non-measurement}. Write $r_a=\mathrm{rank}(R_a)$ and $r=\sum_a r_a$. If $r \leq d$, take $Q_a=R_a$. Otherwise,
		\begin{equation}\label{eq:bound-on-r}
			r
			=
			\sum_a r_a
			=
			\sum_a \Tr(R_a)
			=
			\Tr(R)
			\leq
			(1+2\sqrt{\zeta})d.
		\end{equation}
		Let $\ket{v_{a,1}},\ldots,\ket{v_{a,r_a}}$ be an orthonormal basis for the range of $R_a$, so
		\[
			R_a = \sum_{i=1}^{r_a} \ket{v_{a,i}}\bra{v_{a,i}}.
		\]
		Define the overlaps
		\[
			o_{a,i} = \bra{\psi} (\ket{v_{a,i}}\bra{v_{a,i}} \ot I)\ket{\psi}.
		\]
		Their total satisfies
		\begin{equation}\label{eq:total-overlap}
			\sum_a \sum_{i=1}^{r_a} o_{a,i}
			=
			\bra{\psi} R \ot I \ket{\psi}
			\leq
			1+2\sqrt{\zeta}.
		\end{equation}
		Let $\mathsf{Large}$ be the set of the $d$ largest overlaps $o_{a,i}$, breaking ties arbitrarily, and let $\mathsf{Small}$ be the complement. Then
		\begin{equation}\label{eq:size-of-small-is-small}
			|\mathsf{Small}|
			=
			r-d
			\leq
			2\sqrt{\zeta}\,d
			\leq
			2\sqrt{\zeta}\,r,
		\end{equation}
		by \ref{eq:bound-on-r}. Averaging the overlaps over the $r$ indices gives
		\begin{equation}\label{eq:small-overlaps}
			\sum_{(a,i)\in \mathsf{Small}} o_{a,i}
			\leq
			\frac{|\mathsf{Small}|}{r}\sum_{a,i} o_{a,i}
			\leq
			4\sqrt{\zeta},
		\end{equation}
		using \ref{eq:size-of-small-is-small}, \ref{eq:total-overlap}, and \ref{eq:assumption-on-zeta}.

		For each $a$, let $\mathsf{Large}_a=\{i : (a,i)\in \mathsf{Large}\}$ and define
		\[
			Q_a = \sum_{i \in \mathsf{Large}_a} \ket{v_{a,i}}\bra{v_{a,i}}.
		\]
		Then $\sum_a \mathrm{rank}(Q_a)=|\mathsf{Large}| \leq d$, and $Q_a \leq R_a$ for every $a$, so
		\[
			Q = \sum_a Q_a \leq \sum_a R_a = R \leq (1+2\sqrt{\zeta})I.
		\]
		Also,
		\[
			R_a-Q_a = \sum_{i \in \mathsf{Small}_a} \ket{v_{a,i}}\bra{v_{a,i}}
		\]
		is a projector, hence
		\[
			\sum_a \Vert (R_a-Q_a)\ot I \ket{\psi}\Vert^2
			=
			\sum_{(a,i)\in\mathsf{Small}} o_{a,i}
			\leq
			4\sqrt{\zeta}
		\]
		by \ref{eq:small-overlaps}. Therefore $R_a \ot I \approx_{4\sqrt{\zeta}} Q_a \ot I$. Combining this with \ref{lem:projective-non-measurement} and the triangle inequality for $\approx_\delta$ gives
		\[
			A_a \ot I \approx_{12\sqrt{\zeta}} Q_a \ot I.
		\]
	\end{proof}

	Henceforth let $Q=\{Q_a\}$ be the family given by \ref{lem:projective-low-rank-sum}.

	\begin{lemma}[Completeness of~$Q$]\label{lem:Q-completeness}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.qCompleteness}
		\leanok
		\begin{equation*}
			\bra{\psi} Q \ot I \ket{\psi}
			\geq
			1 - 11 \zeta^{1/4}.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\leanok
		\uses{lem:projective-low-rank-sum}
		Since each $Q_a$ is projective,
		\begin{equation}\label{eq:Q-for-an-A}
			\bra{\psi} Q \ot I \ket{\psi}
			=
			\sum_a \bra{\psi} (Q_a)^2 \ot I \ket{\psi}
			\approx_{5\zeta^{1/4}}
			\sum_a \bra{\psi} (Q_a \cdot A_a)\ot I \ket{\psi}.
		\end{equation}
		By Cauchy--Schwarz,
		\[
			\sum_a \bra{\psi} (Q_a)^2 \ot I \ket{\psi} \leq 1+2\sqrt{\zeta}
			\quad\text{and}\quad
			\sum_a \bra{\psi} (Q_a-A_a)^2 \ot I \ket{\psi} \leq 12\sqrt{\zeta}
		\]
		from \ref{lem:projective-low-rank-sum}.
		A second Cauchy--Schwarz estimate yields
		\begin{equation}\label{eq:another-Q-for-A}
			\sum_a \bra{\psi} (Q_a \cdot A_a)\ot I \ket{\psi}
			\approx_{4\zeta^{1/4}}
			\sum_a \bra{\psi} (A_a)^2 \ot I \ket{\psi}.
		\end{equation}
		Combining \ref{eq:Q-for-an-A} and \ref{eq:another-Q-for-A} with \ref{eq:A-looks-projective} gives
		\[
			\bra{\psi} Q \ot I \ket{\psi}
			\geq
			\sum_a \bra{\psi} A_a \ot I \ket{\psi} - 2\zeta - 9\zeta^{1/4}
			=
			1 - 11\zeta^{1/4},
		\]
		since $\zeta \leq \zeta^{1/4}$ under \ref{eq:assumption-on-zeta}.
	\end{proof}

	\begin{lemma}[Completeness of~$\sqrt{Q}$]\label{lem:sqrt-Q-completeness}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.sqrtQCompleteness}
		\leanok
		\begin{equation*}
			\bra{\psi} \sqrt{Q} \ot I \ket{\psi}
			\geq
			1 - 12 \zeta^{1/4}.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\leanok
		\uses{lem:Q-completeness, lem:projective-low-rank-sum}
		Diagonalize $Q=\sum_i \nu_i \ket{u_i}\bra{u_i}$. Because $Q \leq (1+2\sqrt{\zeta})I$ by \ref{lem:projective-low-rank-sum}, each $\nu_i$ is at most $1+2\sqrt{\zeta}$. Therefore
		\[
			\sqrt{Q}
			=
			\sum_i \sqrt{\nu_i}\ket{u_i}\bra{u_i}
			\geq
			\frac{1}{\sqrt{1+2\sqrt{\zeta}}} Q.
		\]
		The scalar estimate
		\[
			\frac{1}{\sqrt{1+2\sqrt{\zeta}}} \geq 1-\sqrt{\zeta}
		\]
		gives $\sqrt{Q} \geq (1-\sqrt{\zeta})Q$. Hence \ref{lem:Q-completeness} implies
		\[
			\bra{\psi} \sqrt{Q} \ot I \ket{\psi}
			\geq
			(1-\sqrt{\zeta})(1-11\zeta^{1/4})
			\geq
			1-12\zeta^{1/4}.
		\]
	\end{proof}

	\begin{lemma}[$Q$ is almost projective]\label{lem:q-almost-projective}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.qAlmostProjective}
		\leanok
		\begin{equation*}
			\sum_a (Q_a \cdot Q \cdot Q_a - Q_a) \leq 4\sqrt{\zeta} \cdot I.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\leanok
		\uses{lem:projective-low-rank-sum}
		Since $Q \leq (1+2\sqrt{\zeta})I$ by \ref{lem:projective-low-rank-sum},
		\begin{align*}
			\sum_a Q_a \cdot Q \cdot Q_a - \sum_a Q_a
			 & \leq
			(1+2\sqrt{\zeta}) \sum_a Q_a \cdot I \cdot Q_a - \sum_a Q_a \\
			 & =
			(1+2\sqrt{\zeta})Q - Q                                      \\
			 & =
			2\sqrt{\zeta}\,Q                                            \\
			 & \leq
			2\sqrt{\zeta}(1+2\sqrt{\zeta})I                             \\
			 & \leq
			4\sqrt{\zeta}\,I,
		\end{align*}
		using \ref{eq:assumption-on-zeta} in the last step.
	\end{proof}

	\begin{definition}[Matrix decomposition of~$Q_a$]\label{def:matrix-decomposition-Q}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.matrixDecompositionQ}
		\leanok
		\uses{lem:projective-low-rank-sum}
		For each $a$, let
		\[
			m_a = \mathrm{rank}(Q_a),
		\]
		and let $\ket{v_{a,1}},\ldots,\ket{v_{a,m_a}}$ be an orthonormal basis for the range of $Q_a$, so that
		\[
			Q_a = \sum_{i=1}^{m_a} \ket{v_{a,i}}\bra{v_{a,i}}.
		\]
		Let
		\[
			m = \sum_a m_a,
		\]
		and let $\{\ket{a,i}\}$ be an orthonormal basis of $\C^m$, indexed by all $a$ and all
		$1 \leq i \leq m_a$. For each $a$, define
		\[
			X_a = \sum_{i=1}^{m_a} \ket{a,i}\bra{v_{a,i}},
			\qquad
			T_a = \sum_{i=1}^{m_a} \ket{a,i}\bra{a,i},
		\]
		and define
		\begin{equation}\label{eq:looks-like-singular-value-decomposition}
			X = \sum_a X_a = \sum_a \sum_{i=1}^{m_a} \ket{a,i}\bra{v_{a,i}}.
		\end{equation}
		Let $T = \{T_a\}$. Then $T$ is a projective measurement on $\C^m$.
	\end{definition}

	\begin{definition}[SVD of~$X$]\label{def:svd-of-X}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.svdOfX}
		\leanok
		\uses{def:matrix-decomposition-Q}
		Let
		\[
			X = U \cdot \Sigma_{m \times d} \cdot V^\dagger
		\]
		be the singular value decomposition of $X$. Thus $U$ is an $m \times m$ unitary matrix,
		$V$ is a $d \times d$ unitary matrix, and $\Sigma_{m \times d}$ is an
		$m \times d$ diagonal matrix with nonnegative real entries on its diagonal.

		For positive integers $h$ and $w$, write $I_{h \times w}$ for the $h \times w$ matrix
		with $1$'s on its diagonal and $0$'s elsewhere. For integers $h,w \geq m$, write
		$\Sigma_{h \times w}$ for the $h \times w$ diagonal matrix whose diagonal agrees with
		that of $\Sigma_{m \times d}$ and is zero elsewhere.
	\end{definition}

	\begin{remark}\label{rem:motivation-for-P}
		If the family $\{Q_a\}$ were a projective measurement, then the vectors
		$\ket{v_{a,i}}$, over all $a$ and all $1 \leq i \leq m_a$, would be orthonormal.
		Then \eqref{eq:looks-like-singular-value-decomposition} would already be a singular value
		decomposition of $X$, with $\Sigma = I_{m \times d}$. Equivalently,
		\[
			X = U \cdot I_{m \times d} \cdot V^\dagger.
		\]
		This motivates replacing $\Sigma$ by $I$ in the definition of $\widehat X$ and $P$.
	\end{remark}

	\begin{definition}[Definition of~$P$]\label{def:projective-P}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.projectiveP}
		\leanok
		\uses{def:matrix-decomposition-Q, def:svd-of-X}
		Define
		\[
			\widehat{X} = U \cdot I_{m \times d} \cdot V^\dagger.
		\]
		For each $a$, define
		\[
			\widehat{X}_a = T_a \cdot \widehat{X},
			\qquad
			P_a = \widehat{X}_a^\dagger \cdot \widehat{X}_a.
		\]
	\end{definition}

	\begin{lemma}\label{lem:xa-t}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.xa_t}
		\leanok
		For each $a$,
		\[
			X_a = T_a \cdot X.
		\]
	\end{lemma}
	\begin{proof}
		Multiply the definition of $X$ by $T_a$ on the left and use that $T_a$ selects exactly the basis vectors $\ket{a,i}$:
		\[
			T_a \cdot X
			=
			\Big(\sum_{i=1}^{m_a} \ket{a,i}\bra{a,i}\Big)
			\Big(\sum_b \sum_{j=1}^{m_b} \ket{b,j}\bra{v_{b,j}}\Big)
			=
			\sum_{i=1}^{m_a} \ket{a,i}\bra{v_{a,i}}
			=
			X_a.
		\]
	\end{proof}

	\begin{lemma}[$Q_a$ restated]\label{lem:qa-restated}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.qaRestated}
		\leanok
		For each $a$,
		\begin{equation*}
			Q_a = X_a^\dagger \cdot X_a = X^\dagger \cdot T_a \cdot X = X_a^\dagger \cdot X.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\uses{lem:xa-t}
		The identity $X_a^\dagger X_a = Q_a$ follows by expanding the basis sums. The remaining equalities come from \ref{lem:xa-t} and the projectivity of $T$:
		\[
			X_a^\dagger \cdot X_a
			=
			(X^\dagger \cdot T_a)(T_a \cdot X)
			=
			X^\dagger \cdot T_a \cdot X
			=
			X_a^\dagger \cdot X.
		\]
	\end{proof}

	\begin{lemma}[$X$ squared]\label{lem:X-squared}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.xSquared}
		\leanok
		\begin{equation*}
			X \cdot X^\dagger = U \cdot (\Sigma_{m \times m})^2 \cdot U^\dagger,
			\quad
			\text{and}
			\quad
			X^\dagger \cdot X = Q = V \cdot (\Sigma_{d \times d})^2 \cdot V^\dagger.
		\end{equation*}
	\end{lemma}
	% No proof-level marker here: the Lean declaration only projects the stored
	% singular-value data of the chosen $X/\widehat X/P$ witness.
	\begin{proof}
		\uses{lem:qa-restated}
		The first identity is the standard computation from the singular value decomposition:
		\[
			X \cdot X^\dagger
			=
			(U \cdot \Sigma_{m \times d} \cdot V^\dagger)(U \cdot \Sigma_{m \times d} \cdot V^\dagger)^\dagger
			=
			U \cdot (\Sigma_{m \times m})^2 \cdot U^\dagger.
		\]
		For the second, use that $T$ is a measurement and then \ref{lem:qa-restated}:
		\[
			X^\dagger \cdot X
			=
			\sum_a X^\dagger \cdot T_a \cdot X
			=
			\sum_a Q_a
			=
			Q.
		\]
		Computing again from the singular value decomposition gives
		\[
			X^\dagger \cdot X
			=
			V \cdot (\Sigma_{d \times d})^2 \cdot V^\dagger.
		\]
	\end{proof}

	\begin{lemma}\label{lem:X-expression-to-Q-expression}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.xExpressionToQExpression}
		\leanok
		For each $a$,
		\begin{equation*}
			X_a^\dagger \cdot (X \cdot X^\dagger - I_{m \times m})^2 \cdot X_a
			=
			Q_a \cdot Q \cdot Q_a - Q_a.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\uses{lem:X-squared, lem:qa-restated}
		Expand the square and apply \ref{lem:X-squared} and \ref{lem:qa-restated} term by term:
		\[
			X_a^\dagger (X \cdot X^\dagger \cdot X \cdot X^\dagger) X_a
			=
			Q_a \cdot Q \cdot Q_a,
		\]
		while
		\[
			X_a^\dagger (X \cdot X^\dagger) X_a
			=
			(X_a^\dagger \cdot X)(X^\dagger \cdot X_a)
			=
			Q_a \cdot Q_a
			=
			Q_a,
		\]
		and $X_a^\dagger I_{m \times m} X_a = X_a^\dagger X_a = Q_a$. Substituting these identities into
		\[
			(X \cdot X^\dagger - I_{m \times m})^2
			=
			X \cdot X^\dagger \cdot X \cdot X^\dagger - 2X \cdot X^\dagger + I_{m \times m}
		\]
		gives the claim.
	\end{proof}

	\begin{lemma}[$P_a$ restated]\label{lem:pa-restated}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.paRestated}
		\leanok
		For each $a$,
		\begin{equation*}
			P_a = \widehat{X}^\dagger \cdot T_a \cdot \widehat{X} = \widehat{X}_a^\dagger \cdot \widehat{X}.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\uses{def:projective-P}
		Since $T$ is projective,
		\[
			P_a
			=
			\widehat{X}_a^\dagger \cdot \widehat{X}_a
			=
			(\widehat{X}^\dagger \cdot T_a)(T_a \cdot \widehat{X})
			=
			\widehat{X}^\dagger \cdot T_a \cdot \widehat{X}
			=
			\widehat{X}_a^\dagger \cdot \widehat{X}.
		\]
	\end{proof}

	\begin{lemma}[$\widehat{X}$ squared]\label{lem:X-hat-squared}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.xHatSquared}
		\leanok
		\begin{equation*}
			\widehat{X} \cdot \widehat{X}^\dagger = I_{m \times m}.
		\end{equation*}
	\end{lemma}
	% No proof-level marker here: the Lean declaration only projects the stored
	% coisometry identity of $\widehat X$.
	\begin{proof}
		Multiply out the definition of $\widehat X$ and use the unitarity of $U$ and $V$:
		\[
			\widehat{X} \cdot \widehat{X}^\dagger
			=
			(U \cdot I_{m \times d} \cdot V^\dagger)(U \cdot I_{m \times d} \cdot V^\dagger)^\dagger
			=
			U \cdot I_{m \times m} \cdot U^\dagger
			=
			I_{m \times m}.
		\]
	\end{proof}

	\begin{lemma}[$X$ times $\widehat{X}$]\label{lem:X-times-X-hat}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.xTimesXHat}
		\leanok
		\begin{equation*}
			X \cdot \widehat{X}^\dagger = U \cdot \Sigma_{m \times m} \cdot U^\dagger,
			\quad
			\text{and}
			\quad
			X^\dagger \cdot \widehat{X} = \sqrt{Q}.
		\end{equation*}
	\end{lemma}
	% No proof-level marker here: the Lean declaration only projects the stored
	% mixed-product identities of the chosen $X/\widehat X/P$ witness.
	\begin{proof}
		\uses{lem:X-squared}
		The first identity follows directly from the singular value decomposition:
		\[
			X \cdot \widehat{X}^\dagger
			=
			(U \cdot \Sigma_{m \times d} \cdot V^\dagger)(U \cdot I_{m \times d} \cdot V^\dagger)^\dagger
			=
			U \cdot \Sigma_{m \times m} \cdot U^\dagger.
		\]
		Similarly,
		\[
			X^\dagger \cdot \widehat{X}
			=
			(U \cdot \Sigma_{m \times d} \cdot V^\dagger)^\dagger (U \cdot I_{m \times d} \cdot V^\dagger)
			=
			V \cdot \Sigma_{d \times d} \cdot V^\dagger.
		\]
		By \ref{lem:X-squared}, this is the positive square root of
		\[
			V \cdot (\Sigma_{d \times d})^2 \cdot V^\dagger = Q,
		\]
		so $X^\dagger \cdot \widehat X = \sqrt Q$.
	\end{proof}

	\begin{lemma}[Squared difference]\label{lem:squared-difference}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.squaredDifference}
		\leanok
		\begin{equation*}
			(X - \widehat{X}) \cdot (X - \widehat{X})^\dagger \leq (X \cdot X^\dagger - I_{m \times m})^2.
		\end{equation*}
	\end{lemma}
	\begin{proof}
		\uses{lem:X-squared, lem:X-hat-squared, lem:X-times-X-hat}
		By \ref{lem:X-squared}, \ref{lem:X-hat-squared}, and \ref{lem:X-times-X-hat},
		\[
			(X-\widehat X)(X-\widehat X)^\dagger
			=
			U \cdot (\Sigma_{m \times m}-I_{m \times m})^2 \cdot U^\dagger.
		\]
		Since $\Sigma_{m \times m}$ commutes with $I_{m \times m}$,
		\[
			(\Sigma_{m \times m}-I_{m \times m})^2
			\leq
			(\Sigma_{m \times m}^2-I_{m \times m})^2.
		\]
		Conjugating by $U$ and using \ref{lem:X-squared} gives
		\[
			(X-\widehat X)(X-\widehat X)^\dagger
			\leq
			(X \cdot X^\dagger - I_{m \times m})^2.
		\]
	\end{proof}

	\begin{lemma}[Projectivity of~$P$]\label{lem:P-projectivity}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.pProjectivity}
		\leanok
		$P = \{P_a\}$ forms a projective sub-measurement.
	\end{lemma}
	\begin{proof}
		\uses{lem:pa-restated, lem:X-hat-squared}
		For outcomes $a,b$,
		\begin{align*}
			P_a \cdot P_b
			 & =
			(\widehat{X}^\dagger \cdot T_a \cdot \widehat{X})
			(\widehat{X}^\dagger \cdot T_b \cdot \widehat{X})                              \\
			 & =
			\widehat{X}^\dagger \cdot T_a \cdot I_{m \times m} \cdot T_b \cdot \widehat{X} \\
			 & =
			\widehat{X}^\dagger \cdot T_a \cdot \widehat{X} \cdot \bone[a=b]               \\
			 & =
			P_a \cdot \bone[a=b],
		\end{align*}
		where the second line uses \ref{lem:X-hat-squared}, the third uses that $T$ is projective, and the last uses \ref{lem:pa-restated}. Summing over $a$ shows $\sum_a P_a \leq I$, so $P$ is a projective sub-measurement.
	\end{proof}

	\begin{lemma}[$P$ is close to~$Q$]\label{lem:P-Q-approx}
		\lean{MIPStarRE.LDT.MakingMeasurementsProjective.pQApprox}
		\leanok
		\begin{equation*}
			Q_a \ot I \approx_{30\zeta^{1/4}} P_a \ot I.
		\end{equation*}
	\end{lemma}
	% No proof-level marker here: the Lean declaration only projects the stored
	% comparison bound between $P$ and $Q$ from the chosen witness.
	\begin{proof}
		\uses{lem:projective-low-rank-sum, lem:q-almost-projective, lem:sqrt-Q-completeness, lem:xa-t, lem:qa-restated, lem:X-expression-to-Q-expression, lem:pa-restated, lem:X-hat-squared, lem:X-times-X-hat, lem:squared-difference, lem:P-projectivity}
		The quantity to bound is
		\begin{equation}\label{eq:P-Q-thing-to-bound}
			\sum_a \bra{\psi} (Q_a-P_a)^2 \ot I \ket{\psi}
			=
			\bra{\psi} Q \ot I \ket{\psi}
			+
			\bra{\psi} P \ot I \ket{\psi}
			-
			\sum_a \bra{\psi} Q_aP_a \ot I \ket{\psi}
			-
			\sum_a \bra{\psi} P_aQ_a \ot I \ket{\psi},
		\end{equation}
		using the projectivity of $Q$ and $P$. The first term is at most $1+2\sqrt{\zeta}$ by \ref{lem:projective-low-rank-sum}, and the second is at most $1$ because $P$ is a sub-measurement.

		For the last two terms,
		\begin{equation}\label{eq:complex-conjugates}
			\sum_a \bra{\psi} Q_aP_a \ot I \ket{\psi}
			+
			\sum_a \bra{\psi} P_aQ_a \ot I \ket{\psi}
			=
			2 \cdot \mathfrak{R}\Big(\sum_a \bra{\psi} Q_aP_a \ot I \ket{\psi}\Big).
		\end{equation}
		By \ref{lem:qa-restated},
		\[
			\sum_a \bra{\psi} Q_aP_a \ot I \ket{\psi}
			=
			\sum_a \bra{\psi} (X_a^\dagger \cdot X \cdot P_a) \ot I \ket{\psi}.
		\]
		The main estimate is
		\begin{equation}\label{eq:add-a-hat}
			\sum_a \bra{\psi} (X_a^\dagger \cdot X \cdot P_a)\ot I \ket{\psi}
			\approx_{2\zeta^{1/4}}
			\sum_a \bra{\psi} (X_a^\dagger \cdot \widehat{X} \cdot P_a)\ot I \ket{\psi}.
		\end{equation}
		This follows by Cauchy--Schwarz:
		\[
			\Big|\sum_a \bra{\psi} ((X_a^\dagger \cdot (X-\widehat X)) \ot I)\cdot(P_a \ot I)\ket{\psi}\Big|
		\]
		is bounded by the square root of
		\[
			\sum_a \bra{\psi} \big(X_a^\dagger \cdot (X-\widehat X)(X-\widehat X)^\dagger \cdot X_a\big)\ot I \ket{\psi},
		\]
		times the square root of $\sum_a \bra{\psi}(P_a)^2 \ot I \ket{\psi} \leq 1$. The first factor is at most $4\sqrt{\zeta}$ by \ref{lem:squared-difference}, \ref{lem:X-expression-to-Q-expression}, and \ref{lem:q-almost-projective}, so the total error is at most $2\zeta^{1/4}$.

		The right-hand side of \ref{eq:add-a-hat} is real, because
		\[
			\sum_a \bra{\psi} (X_a^\dagger \cdot \widehat{X} \cdot P_a)\ot I \ket{\psi}
			=
			\bra{\psi} (X^\dagger \cdot \widehat{X}) \ot I \ket{\psi}
			=
			\bra{\psi} \sqrt{Q} \ot I \ket{\psi},
		\]
		where the first equality uses \ref{lem:xa-t}, \ref{lem:pa-restated}, and \ref{lem:X-hat-squared} and that $T$ is a measurement, and the second uses \ref{lem:X-times-X-hat}. Hence \ref{lem:sqrt-Q-completeness} and \ref{eq:add-a-hat} imply
		\[
			\mathfrak{R}\Big(\sum_a \bra{\psi} Q_aP_a \ot I \ket{\psi}\Big)
			\geq
			1-14\zeta^{1/4}.
		\]
		Substituting this into \ref{eq:complex-conjugates} and then into \ref{eq:P-Q-thing-to-bound} gives
		\[
			\sum_a \bra{\psi} (Q_a-P_a)^2 \ot I \ket{\psi}
			\leq
			(1+2\sqrt{\zeta}) + 1 - 2(1-14\zeta^{1/4})
			=
			2\sqrt{\zeta} + 28\zeta^{1/4}
			\leq
			30\zeta^{1/4}.
		\]
	\end{proof}

	Finally, \ref{lem:projective-low-rank-sum} gives
	\[
		A_a \ot I \approx_{12\sqrt{\zeta}} Q_a \ot I,
	\]
	and \ref{lem:P-Q-approx} gives
	\[
		Q_a \ot I \approx_{30\zeta^{1/4}} P_a \ot I.
	\]
	The triangle inequality for $\approx_\delta$ yields
	\[
		A_a \ot I \approx_{84\zeta^{1/4}} P_a \ot I.
	\]
\end{proof}
