\begin{lemma}[Replacing a line answer by a fixed polynomial]\label{lem:generalize-b}
	\lean{MIPStarRE.LDT.GlobalVariance.generalizeB, MIPStarRE.LDT.GlobalVariance.generalizeBCollisionResidual, MIPStarRE.LDT.GlobalVariance.generalizeBCollisionResidual_eq_seedCollisionExpansion, MIPStarRE.LDT.GlobalVariance.generalizeBCollisionResidual_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBPointwiseSchwartzZippel, MIPStarRE.LDT.GlobalVariance.generalizeBFromSchwartzZippel, MIPStarRE.LDT.GlobalVariance.generalizeBSeedCollisionExpansion, MIPStarRE.LDT.GlobalVariance.generalizeBSeedCollisionExpansion_eq_lineCollisionExpansion, MIPStarRE.LDT.GlobalVariance.generalizeBSeedCollisionExpansion_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBLineCollisionExpansion, MIPStarRE.LDT.GlobalVariance.generalizeBLineCollisionExpansion_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBReversePointwiseBound, MIPStarRE.LDT.GlobalVariance.generalizeBReversePointwiseBound_polysum_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBLineCollisionExpansion_polysum_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBSeedCollisionExpansion_polysum_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBCollisionResidual_polysum_le_error, MIPStarRE.LDT.GlobalVariance.generalizeBDeviationAtPolynomial_polysum_le_error}
	\leanok
	\uses{def:polymeasurements, lem:schwartz-zippel-individual, lem:axis-line-polynomial-agreement-mdq}
	Let $G \in \polysub{m}{q}{d}$. On the axis-parallel line-test distribution,
	\[
		B^{\ell}_{[f(u)=g(u)]} \ot (G_g)^{1/2}
		\approx_{md/q}
		B^{\ell}_{g|_{\ell}} \ot (G_g)^{1/2}.
	\]
\end{lemma}
\begin{proof}
	\leanok
	\uses{lem:schwartz-zippel-individual, lem:axis-line-polynomial-agreement-mdq}
	Expand the $\approx_{md/q}$ norm on the weighted vector $(I\ot (G_g)^{1/2})\ket{\psi}$. Because $B^\ell$ is a projective measurement, only those line polynomials $f$ with $f\ne g|_\ell$ and $f(u)=g(u)$ contribute. Lemma~\ref{lem:axis-line-polynomial-agreement-mdq} is the univariate Schwartz--Zippel wrapper used in Lean for this line-restriction event; it supplies the paper's $md/q$ loss, and the remaining operator sum is bounded by the fact that $G$ is a submeasurement.
\end{proof}

\begin{lemma}[Local variance of the points measurements]\label{lem:local-variance-of-points}
	\lean{MIPStarRE.LDT.GlobalVariance.localVarianceOfPoints, MIPStarRE.LDT.GlobalVariance.localVarianceTransportChainError, MIPStarRE.LDT.GlobalVariance.localVarianceTransportChainError_le_localVarianceOfPointsError, MIPStarRE.LDT.GlobalVariance.localVarianceTransportChainBound, MIPStarRE.LDT.GlobalVariance.localVarianceOfPointsFromTransportChainBound, MIPStarRE.LDT.GlobalVariance.globalVarianceOfPointsFromTransportChainBound}
	\leanok
	\uses{lem:good-strategy-characterization, prop:simeq-to-approx, lem:generalize-b, prop:triangle-inequality-for-approx_delta}
	Let $G \in \polysub{m}{q}{d}$. On the hypercube edge distribution $(u,v) \sim C$,
	\begin{equation}
		A^u_{g(u)} \ot (G_g)^{1/2}
		\approx_{24\cdot\left(\eps+\delta+\frac{md}{q}\right)}
		A^v_{g(v)} \ot (G_g)^{1/2}.
		\label{eq:local-variance-of-points-equation}
	\end{equation}
\end{lemma}
\begin{proof}\leanok
	\uses{prop:simeq-to-approx, prop:cab-approx-delta, lem:generalize-b, lem:good-strategy-characterization, prop:triangle-inequality-for-approx_delta}
	Fix $g$ and keep the weight $(G_g)^{1/2}$ on the second tensor factor throughout. On the vector $(I\ot (G_g)^{1/2})\ket{\psi}$ one has the six-step chain
	\[
		A^u_{g(u)} \ot (G_g)^{1/2}
		\approx_{2\delta} I\ot (G_g)^{1/2} A^u_{g(u)}
		\approx_{2\eps} B^\ell_{[f(u)=g(u)]} \ot (G_g)^{1/2}
		\approx_{md/q} B^\ell_{g|_\ell} \ot (G_g)^{1/2}
		\approx_{md/q} B^\ell_{[f(v)=g(v)]} \ot (G_g)^{1/2}
		\approx_{2\eps} I\ot (G_g)^{1/2} A^v_{g(v)}
		\approx_{2\delta} A^v_{g(v)} \ot (G_g)^{1/2}.
	\]
	The first, second, fifth, and sixth comparisons use Lemma~\ref{prop:simeq-to-approx} together with Theorem~\ref{prop:cab-approx-delta} and the good-strategy hypotheses, and the middle pair is Lemma~\ref{lem:generalize-b}. Lemma~\ref{prop:triangle-inequality-for-approx_delta} sums these six weighted transports.
\end{proof}

\begin{lemma}[Equivalent local-variance bound]\label{lem:equivalent-local-variance}
	\lean{MIPStarRE.LDT.GlobalVariance.localVarianceDeviation_sum_le_localVarianceOfPointsError}
	\leanok
	\uses{lem:local-variance-of-points}
	The local-variance bound for the point measurements is
	\begin{equation}
		\sum_{g \in \polyfunc{m}{q}{d}} \E_{(u,v)\sim C} \bra{\psi} (A^u_{g(u)}-A^v_{g(v)})^2 \ot G_g \ket{\psi}
		\le
		24\left(\eps+\delta+\frac{md}{q}\right).
		\label{eq:equivalent-local-variance}
	\end{equation}
\end{lemma}

\begin{remark}[Polynomial-sum formalization for \ref{eq:equivalent-local-variance}]\label{rem:equivalent-local-variance-formalization-inputs}
	\lean{MIPStarRE.LDT.GlobalVariance.axisParallelBaseEventApproximation_weighted_sample_sum, MIPStarRE.LDT.GlobalVariance.axisParallelPointLineConsistency_weighted_leftToRightLineQuestion_sum, MIPStarRE.LDT.GlobalVariance.axisParallelPointLineConsistency_weighted_rightToLeftLineQuestion_sum}
	\uses{lem:equivalent-local-variance, lem:local-variance-of-points, lem:generalize-b}
	The three referenced declarations and Lemma~\ref{lem:equivalent-local-variance} together provide the full polynomial-sum (cardinality-free) chain proving the equation.  The chain assembly lemma \texttt{localVarianceDeviation\_sum\_le\_localVarianceOfPointsError} telescopes the six operator differences via \texttt{ev\_sum\_conjTranspose\_mul\_sum\_le}, sums over~$g$, and applies the six individual \texttt{\_sum} bounds (including the reverse~\texttt{generalizeB} sum) to avoid the per-polynomial cardinality blow-up.
\end{remark}

\begin{lemma}[Global variance of the points measurements]\label{lem:global-variance-of-points}
	\lean{MIPStarRE.LDT.GlobalVariance.globalVarianceOfPoints, MIPStarRE.LDT.GlobalVariance.globalVarianceDeviation_sum_le_m_mul_localVarianceDeviation_sum, MIPStarRE.LDT.GlobalVariance.globalVarianceDeviation_sum_le_of_localVarianceDeviation_sum_le}
	\leanok
	\uses{def:local-and-variance, lem:local-to-global, lem:equivalent-local-variance}
	Let $G \in \polysub{m}{q}{d}$. On the uniform distribution over independent $u,v \in \F_q^m$,
	\begin{equation}
		A^u_{g(u)} \ot (G_g)^{1/2}
		\approx_{24m\left(\eps+\delta+\frac{md}{q}\right)}
		A^v_{g(v)} \ot (G_g)^{1/2}.
		\label{eq:global-variance-of-points-equation}
	\end{equation}
\end{lemma}
\begin{proof}
	\leanok
	\uses{def:local-and-variance, lem:local-to-global, lem:local-variance-of-points, lem:equivalent-local-variance}
	We want to bound
	\begin{equation}
		\E_{u,v \sim \F_q^m} \sum_{g \in \polyfunc{m}{q}{d}} \lVert (A^u_{g(u)}-A^v_{g(v)}) \ot (G_g)^{1/2} \ket{\psi} \rVert^2
		=
		\E_{u,v \sim \F_q^m} \sum_{g \in \polyfunc{m}{q}{d}} \bra{\psi} (A^u_{g(u)}-A^v_{g(v)})^2 \ot G_g \ket{\psi}.
		\label{eq:global-variance-target}
	\end{equation}
	For each $g \in \polyfunc{m}{q}{d}$, define
	\[
		A(g)^u := A^u_{g(u)},
		\qquad
		\ket{\psi_g} := (I \ot (G_g)^{1/2})\ket{\psi}.
	\]
	Then
	\[
		\eqref{eq:global-variance-target}
		=
		\sum_{g \in \polyfunc{m}{q}{d}} \E_{u,v \sim \F_q^m} \bra{\psi_g} (A(g)^u-A(g)^v)^2 \ot I \ket{\psi_g}
		=
		\sum_{g \in \polyfunc{m}{q}{d}} 2\,\mathbf{Var}_{\mathrm{global}}(A(g),\psi_g).
	\]
	Applying Lemma~\ref{lem:local-to-global} gives
	\[
		\eqref{eq:global-variance-target}
		\le
		\sum_{g \in \polyfunc{m}{q}{d}} 2m\,\mathbf{Var}_{\mathrm{local}}(A(g),\psi_g)
		=
		m \cdot \sum_{g \in \polyfunc{m}{q}{d}} \E_{(u,v)\sim C} \bra{\psi} (A^u_{g(u)}-A^v_{g(v)})^2 \ot G_g \ket{\psi}.
	\]
	The last quantity is at most $m \cdot 24\left(\eps+\delta+\frac{md}{q}\right)$ by Lemma~\ref{lem:equivalent-local-variance}, which is exactly \ref{eq:global-variance-of-points-equation}.
\end{proof}
