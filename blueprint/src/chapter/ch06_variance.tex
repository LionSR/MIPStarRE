\chapter{Global variance of the points measurements}
\label{chap:variance}

Throughout this chapter, $(\psi,A,B,L)$ is an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m,q,d)$ low individual degree test.


\begin{lemma}[Replacing a line answer by a fixed polynomial]\label{lem:generalize-b}
  \lean{MIPStarRE.Paper2009LDT.Section8GlobalVariance.generalizeB}
  \uses{def:polymeasurements, lem:schwartz-zippel-individual}
  Let $G \in \polysub{m}{q}{d}$. On the axis-parallel line-test distribution,
  \[
    B^{\ell}_{[f(u)=g(u)]} \ot (G_g)^{1/2}
    \approx_{md/q}
    B^{\ell}_{g|_{\ell}} \ot (G_g)^{1/2}.
  \]
\end{lemma}
\begin{proof}
  Expand the $\approx_{md/q}$ norm on the weighted vector $(I\ot (G_g)^{1/2})\ket{\psi}$. Because $B^\ell$ is a projective measurement, only those line polynomials $f$ with $f\ne g|_\ell$ and $f(u)=g(u)$ contribute. Lemma~\ref{lem:schwartz-zippel-individual} bounds the probability of this event by $md/q$, and the remaining operator sum is bounded by the fact that $G$ is a submeasurement.
\end{proof}

\begin{lemma}[Local variance of the points measurements]\label{lem:local-variance-of-points}
  \lean{MIPStarRE.Paper2009LDT.Section8GlobalVariance.localVarianceOfPoints}
  \uses{lem:good-strategy-characterization, prop:simeq-to-approx, lem:generalize-b, prop:triangle-inequality-for-approx_delta}
  Let $G \in \polysub{m}{q}{d}$. On the hypercube edge distribution $(u,v) \sim C$,
  \[
    A^u_{g(u)} \ot (G_g)^{1/2}
    \approx_{24\left(\eps+\delta+\frac{md}{q}\right)}
    A^v_{g(v)} \ot (G_g)^{1/2}.
  \]
\end{lemma}
\begin{proof}
  Fix $g$ and keep the weight $(G_g)^{1/2}$ on the second tensor factor throughout. On the vector $(I\ot (G_g)^{1/2})\ket{\psi}$ one has the six-step chain
  \[
    A^u_{g(u)} \ot (G_g)^{1/2}
    \approx_{2\delta} I\ot (G_g)^{1/2} A^u_{g(u)}
    \approx_{2\eps} B^\ell_{[f(u)=g(u)]} \ot (G_g)^{1/2}
    \approx_{md/q} B^\ell_{g|_\ell} \ot (G_g)^{1/2},
  \]
  followed by the same three steps in reverse with $u$ replaced by $v$. The first, second, fifth, and sixth comparisons are the consistency-to-distance transports supplied by Lemma~\ref{prop:simeq-to-approx}; the middle pair is Lemma~\ref{lem:generalize-b}. Lemma~\ref{prop:triangle-inequality-for-approx_delta} sums these six weighted transports.
\end{proof}

\begin{lemma}[Global variance of the points measurements]\label{lem:global-variance-of-points}
  \lean{MIPStarRE.Paper2009LDT.Section8GlobalVariance.globalVarianceOfPoints}
  \uses{def:local-and-variance, lem:local-to-global, lem:local-variance-of-points}
  Let $G \in \polysub{m}{q}{d}$. On the uniform distribution over independent $u,v \in \F_q^m$,
  \[
    A^u_{g(u)} \ot (G_g)^{1/2}
    \approx_{24m\left(\eps+\delta+\frac{md}{q}\right)}
    A^v_{g(v)} \ot (G_g)^{1/2}.
  \]
\end{lemma}
\begin{proof}
  For each polynomial $g$, set $A(g)^u=A^u_{g(u)}$ and work with the weighted state $\ket{\psi_g}=(I \ot (G_g)^{1/2})\ket{\psi}$. Lemma~\ref{lem:local-variance-of-points} is exactly the local-variance bound for the family $u \mapsto A(g)^u$ on $\ket{\psi_g}$, and Lemma~\ref{lem:local-to-global} transports that bound from hypercube edges to independent points.
\end{proof}
