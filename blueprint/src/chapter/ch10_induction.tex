\section{Restricting to one slice}

\begin{definition}[Appending a line at height $x$]\label{def:append-x}
	\lean{MIPStarRE.LDT.AxisParallelLine.appendAtHeight, MIPStarRE.LDT.AxisLinePolynomial.appendAtHeight, MIPStarRE.LDT.DiagonalLine.appendAtHeight}
	\leanok
  For $x \in \F_q$, let $\operatorname{append}_x(\ell)$ be the line in $\F_q^{m+1}$ obtained by appending the coordinate $x$ to every point of a line $\ell \subseteq \F_q^m$. If $f \colon \ell \to \F_q$, let $\operatorname{append}_x(f)$ be the polynomial on $\operatorname{append}_x(\ell)$ defined by $\operatorname{append}_x(f)(u,x)=f(u)$.
\end{definition}

\begin{definition}[$x$-restricted strategy]\label{def:restricted-strategy}
	\lean{MIPStarRE.LDT.MainInductionStep.xRestrictedStrategy, MIPStarRE.LDT.MainInductionStep.RestrictedSymStrat}
	\leanok
  \uses{def:append-x, def:symmetric-projective-strategy}
  Let $(\psi,A,B,L)$ be a symmetric strategy for the $(m+1,q,d)$ low individual degree test. For each $x \in \F_q$, the $x$-restricted strategy $(\psi,A^x,B^x,L^x)$ for the $(m,q,d)$ test is defined by
  \[
    (A^x)_a^u = A_a^{u,x},
    \qquad
    (B^x)_f^\ell = B_{\operatorname{append}_x(f)}^{\operatorname{append}_x(\ell)},
    \qquad
    (L^x)_f^\ell = L_{\operatorname{append}_x(f)}^{\operatorname{append}_x(\ell)}.
  \]
\end{definition}

\begin{lemma}[Restricted success probabilities]\label{lem:restricted-probabilities}
  \lean{MIPStarRE.LDT.MainInductionStep.restrictedProbabilities}
	\leanok
  \uses{def:restricted-strategy, def:good-strategy}
  Let $(\psi,A,B,L)$ be an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m+1,q,d)$ test. For each $x \in \F_q$, let $(\psi,A^x,B^x,L^x)$ be the restricted strategy and let $\eps_x,\delta_x,\gamma_x$ be its three failure probabilities. Then
  \[
    \E_x \eps_x \le \frac{m+1}{m}\eps,
    \qquad
    \E_x \delta_x \le \delta,
    \qquad
    \E_x \gamma_x \le \frac{m+1}{m}\gamma.
  \]
\end{lemma}
\begin{proof}
	\leanok
	The self-consistency test restricts exactly to the slice at height $x$. For the axis-parallel and diagonal-line tests, conditioning on the sampled line not being parallel to the last coordinate identifies the test with the corresponding restricted game.
\end{proof}

\begin{definition}[Average restricted errors]\label{def:average-restricted-errors}
	\lean{MIPStarRE.LDT.MainInductionStep.averageRestrictedAxisParallelError, MIPStarRE.LDT.MainInductionStep.averageRestrictedSelfConsistencyError, MIPStarRE.LDT.MainInductionStep.averageRestrictedDiagonalError}
	\leanok
	\uses{def:restricted-strategy}
	The average restricted errors are the expectations over $x \in \F_q$ of the
	axis-parallel, self-consistency, and restricted diagonal-line failure
	probabilities of the $x$-restricted strategies.
\end{definition}

\begin{lemma}[Weighted restricted-probability bounds]\label{lem:weighted-restricted-probability-bounds}
	\lean{MIPStarRE.LDT.MainInductionStep.RestrictedProbabilitiesStatement.ofWeightedBounds}
	\leanok
	\uses{lem:restricted-probabilities, def:average-restricted-errors}
	The weighted bounds from Lemma~\ref{lem:restricted-probabilities} imply the
	restricted-probability data used in the successor step.
\end{lemma}
\begin{proof}
	\leanok
	The weighted axis-parallel and diagonal-line estimates are divided by the
	transverse-direction weight, while the self-consistency estimate is exactly
	the average restricted self-consistency probability. These three estimates
	form the restricted-probability record used in the successor step.
\end{proof}

\section{The inductive construction}

\begin{theorem}[Self-improvement]\label{thm:self-improvement-in-induction-section}
  \lean{MIPStarRE.LDT.MainInductionStep.selfImprovementInInductionSection}
  \leanok
  \uses{def:good-strategy, def:polymeasurements, def:simeq, def:approx_delta}
  Let $(\psi,A,B,L)$ be an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m,q,d)$ low individual degree test. Let $G \in \polymeas{m}{q}{d}$ be a measurement with the following property:
  \begin{enumerate}
    \item (Consistency with~$A$): \label{item:si-inductive-G-consistency} On average over $u \sim \F_q^m$,
    \[
      A_a^u \ot I \simeq_\nu I \ot G_{[g(u)=a]}.
    \]
  \end{enumerate}
  Let
  \[
    \zeta = 3000m\left(\eps^{1/32}+\delta^{1/32}+(d/q)^{1/32}\right).
  \]
  Then there exists a projective submeasurement $H \in \polysub{m}{q}{d}$ with the following properties:
  \begin{enumerate}
    \item (Completeness): \label{item:si-inductive-completeness} If $H=\sum_h H_h$, then
    \[
      \bra{\psi} H \ot I \ket{\psi} \ge (1-\nu)-\zeta.
    \]
    \item (Consistency with~$A$): \label{item:si-inductive-A-consistency} On average over $u \sim \F_q^m$,
    \[
      A_a^u \ot I \simeq_\zeta I \ot H_{[h(u)=a]}.
    \]
    \item (Strong self-consistency): \label{item:si-inductive-self}
    \[
      H_h \ot I \approx_\zeta I \ot H_h.
    \]
    \item (Boundedness): \label{item:si-inductive-boundedness} There exists a positive semidefinite operator $Z$ such that
    \[
      \bra{\psi} Z \ot (I-H) \ket{\psi} \le \zeta
    \]
    and for each $h \in \polyfunc{m}{q}{d}$,
    \[
      Z \ge \E_u A_{h(u)}^u.
    \]
  \end{enumerate}
\end{theorem}
\begin{proof}
	\uses{thm:self-improvement, prop:two-notions-of-self-consistency}
	This is Theorem~\ref{thm:self-improvement}, with the conclusions reformulated in the notation used by the induction step. Since the output is projective, Lemma~\ref{prop:two-notions-of-self-consistency} converts the strong self-consistency statement to the usual bipartite formulation.
\end{proof}

\begin{remark}[Lean realization of induction self-improvement]\label{rem:self-improvement-in-induction-section-realization}
	The Lean declaration for
	Theorem~\ref{thm:self-improvement-in-induction-section} has the paper
	hypotheses, in particular that \(G\) is a complete polynomial measurement.
	The labelled theorem in
	\texttt{references/ldt-paper/inductive\_step.tex:249--286} states the input
	as a submeasurement, while the proved form in
	\texttt{references/ldt-paper/self\_improvement.tex:635--671} states it as a
	measurement.  The blueprint and Lean follow the proved measurement-valued
	form, which is also the form used at the point where the theorem is applied
	in \texttt{references/ldt-paper/inductive\_step.tex:461--485}.
	The proof applies
	\texttt{SelfImprovement.selfImprovement} to \(G\), then uses
	\texttt{selfImprovementInInductionSectionConclusion\_ofSelfImprovementConclusion}
	to transport the fields.
	It therefore inherits the Section~\ref{chap:self-improvement} SDP proof debt:
	\texttt{SelfImprovement.selfImprovement} still depends on
	\texttt{sdp\_statement\_with\_slackness}, whose complementary-slackness proof
	is tracked by issue \#1230.  This is not an additional hypothesis of the
	induction-section theorem.
\end{remark}

\begin{theorem}[Pasting]\label{thm:ld-pasting-in-induction-section}
  \lean{MIPStarRE.LDT.MainInductionStep.ldPastingInInductionSection}
  \leanok
  \uses{def:good-strategy, def:polymeasurements, def:simeq, def:approx_delta}
  Let $(\psi,A,B,L)$ be an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m+1,q,d)$ low individual degree test. Let $\{G^x\}_{x \in \F_q}$ be projective submeasurements in $\polysub{m}{q}{d}$ with the following properties:
  \begin{enumerate}
    \item (Completeness): \label{item:ld-pasting-inductive-completeness} If $G=\E_x \sum_g G_g^x$, then
    \[
      \bra{\psi} G \ot I \ket{\psi} \ge 1-\kappa.
    \]
    \item (Consistency with~$A$): \label{item:ld-pasting-inductive-consistency} On average over $(u,x) \sim \F_q^{m+1}$,
    \[
      A_a^{u,x} \ot I \simeq_\zeta I \ot G_{[g(u)=a]}^x.
    \]
    \item (Strong self-consistency): \label{item:ld-pasting-inductive-self-consistency} On average over $x \sim \F_q$,
    \[
      G_g^x \ot I \approx_\zeta I \ot G_g^x.
    \]
    \item (Boundedness): \label{item:ld-pasting-inductive-boundedness} There are positive semidefinite operators $Z^x$ such that
    \[
      \E_x \bra{\psi} (I-G^x) \ot Z^x \ket{\psi} \le \zeta
    \]
    and for each $x \in \F_q$ and $g \in \polyfunc{m}{q}{d}$,
    \[
      Z^x \ge \E_u A_{g(u)}^{u,x}.
    \]
  \end{enumerate}
  Let $k \ge 400md$ and set
  \[
    \nu = 100k^2m\left(\eps^{1/32}+\delta^{1/32}+\gamma^{1/32}+\zeta^{1/32}+(d/q)^{1/32}\right)
  \]
  and
  \[
    \sigma = \kappa\left(1+\frac{1}{100m}\right) + 2\nu + e^{-k/(80000m^2)}.
  \]
  Then there exists a measurement $H \in \polymeas{m+1}{q}{d}$ with the following property:
  \begin{enumerate}
    \item (Consistency with~$A$): \label{item:ld-pasting-inductive-N-consistency} On average over $u \sim \F_q^{m+1}$,
    \[
      A_a^u \ot I \simeq_\sigma I \ot H_{[h(u)=a]}.
    \]
  \end{enumerate}
\end{theorem}
\begin{proof}
	\leanok
	\uses{thm:ld-pasting}
	This is exactly Theorem~\ref{thm:ld-pasting}, restated in the induction chapter
	because it is the pasting input used in the proof of
	Theorem~\ref{thm:main-induction}.  The Lean declaration linked above applies
	the unrestricted formal pasting theorem and transports its point-consistency
	field to the induction-section conclusion record.
\end{proof}

\begin{remark}[Restricted Lean form of induction-section pasting]\label{rem:ld-pasting-induction-restricted-lean-form}
	The Lean declaration linked here is the restricted nontrivial-regime
	restatement of Theorem~\ref{thm:ld-pasting-in-induction-section}.  Its
	statement calls the formal Section~\ref{chap:pasting} pasting theorem and
	therefore assumes, in addition to the source hypotheses,
	\[
		\gamma \le 1,\qquad \zeta \le 1,\qquad d \le q,\qquad 0<d,\qquad 1\le k.
	\]
	The corresponding paper theorem in
	\texttt{references/ldt-paper/ld-pasting.tex}, lines 12--50, does not include
	these assumptions; lines 52--55 record them only as a proof reduction to the
	nontrivial regime.  The trivial complementary cases remain to be formalized
	before this declaration can be marked as the unrestricted source theorem.
\end{remark}

\begin{lemma}[Base case of main induction]\label{lem:main-induction-base}
	\lean{MIPStarRE.LDT.MainInductionStep.mainInductionBaseCase}
	\leanok
	\uses{def:good-strategy, def:polymeasurements, def:simeq}
	Under the hypotheses of Theorem~\ref{thm:main-induction} with $m=1$, there exists a measurement $G \in \polymeas{1}{q}{d}$ such that, on average over $u \sim \F_q$,
	\[
		A_a^u \ot I \simeq_\eps I \ot G_{[g(u)=a]},
	\]
	which is stronger than the bound required by Theorem~\ref{thm:main-induction} for $m=1$.
\end{lemma}
\begin{proof}
	\leanok
	For $m=1$, there is only one axis-parallel line $\ell$ in $\F_q$, so the line measurement $B^\ell$ is already an element of $\polymeas{1}{q}{d}$. Because the strategy fails the axis-parallel lines test with probability at most $\eps$, the consistency bound follows directly from the axis-parallel consistency hypothesis.
\end{proof}

\begin{theorem}[Main induction]\label{thm:main-induction}
	\lean{MIPStarRE.LDT.MainInductionStep.mainInduction}
	\uses{def:good-strategy, def:polymeasurements, def:simeq, lem:main-induction-base}
  Let $(\psi,A,B,L)$ be an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m,q,d)$ low individual degree test. Let $k \ge 400md$ and set
  \[
    \nu = 1000k^2m^2\left(\eps^{1/1024}+\delta^{1/1024}+\gamma^{1/1024}+(d/q)^{1/1024}\right)
  \]
  and
  \[
    \sigma = m^2\left(\nu + e^{-k/(80000m^2)}\right).
  \]
  Then there exists a measurement $G \in \polymeas{m}{q}{d}$ such that, on average over $u \sim \F_q^m$,
  \[
    A_a^u \ot I \simeq_\sigma I \ot G_{[g(u)=a]}.
  \]
\end{theorem}
\begin{proof}
	\uses{lem:restricted-probabilities, lem:weighted-restricted-probability-bounds, lem:main-induction-base, thm:self-improvement-in-induction-section, thm:ld-pasting-in-induction-section, def:successor-pasting-data}
	In Lean, the base case is formalized through
	\lean{MIPStarRE.LDT.MainInductionStep.mainInductionBaseCase}.
	The successor step of
	\lean{MIPStarRE.LDT.MainInductionStep.mainInduction} is isolated in its
	native \(m\to m+1\) form as the source-shaped proof obligation
	\lean{MIPStarRE.LDT.MainInductionStep.mainInductionSuccessorNext}.  The
	non-base branch
	\lean{MIPStarRE.LDT.MainInductionStep.mainInductionSuccessor} only decomposes
	an arbitrary parameter bundle with \(m\ne1\) into this successor form.  Neither
	declaration takes restricted-probability, recursive-slice, self-improvement, or
	pasting records as hypotheses.  The nearby formalization still names the stage data
	\lean{MIPStarRE.LDT.MainInductionStep.SliceRestrictionData, MIPStarRE.LDT.MainInductionStep.PerSliceInductionData, MIPStarRE.LDT.MainInductionStep.SelfImprovementData, MIPStarRE.LDT.MainInductionStep.AveragedPastingData}
	\lean{MIPStarRE.LDT.MainInductionStep.SelfImprovementData.SliceStrategyTransport, MIPStarRE.LDT.MainInductionStep.SelfImprovementData.ofSliceStrategyTransport}
	\lean{MIPStarRE.LDT.MainInductionStep.SelfImprovementData.SliceStrategyTransport.averagedPoint_eq_of_pointMeasurement_eq, MIPStarRE.LDT.MainInductionStep.SelfImprovementData.SliceStrategyTransport.ofPointMeasurementEq}
	that mirror the four paper stages $\text{restrict} \to \text{induct} \to
	\text{self-improve} \to \text{paste}$, but no theorem with these data as
	additional source-level hypotheses is advertised as
	Theorem~\ref{thm:main-induction}. For the answer-valued restriction route
	used in the successor branch of the main formal theorem, Lean now also exposes
	\lean{MIPStarRE.LDT.MainInductionStep.AnswerMainInductionConclusion, MIPStarRE.LDT.MainInductionStep.AnswerMainInductionHypothesis, MIPStarRE.LDT.MainInductionStep.AnswerPerSliceInductionData.ofMainInductionHypothesis},
	\lean{MIPStarRE.LDT.MainInductionStep.AnswerPerSliceInductionData.ofLegacy, MIPStarRE.LDT.MainInductionStep.PerSliceInductionData.ofAnswer, MIPStarRE.LDT.MainInductionStep.SelfImprovementData.ofAnswerForLegacy, MIPStarRE.LDT.MainInductionStep.SelfImprovementData.ofAnswer},
	which record the predecessor induction hypothesis for the restricted
	strategies, transport induction data in both directions between the
	answer-valued and legacy slice interfaces, and then forget only the
	answer-valued diagonal alphabet before the existing pasting assembly. The weighted
	restricted-probability estimates are now public via
	\lean{MIPStarRE.LDT.MainInductionStep.weighted_axisParallel_bound, MIPStarRE.LDT.MainInductionStep.weighted_diagonal_bound, MIPStarRE.LDT.MainInductionStep.restrictedProbabilities},
	The remaining proof gap for Theorem~\ref{thm:main-induction} is to derive the
	successor-step constructions, including the restricted-strategy
	self-improvement and slice-transport construction
	tracked by issues \#1507 and \#1503 under the bridge-debt tracker \#1458,
	from the corrected theorem hypotheses; hence no proof-level \leanok is attached here.
	We may assume $\eps,\delta,\gamma,d/q \le 1$, since otherwise $\nu \ge 1$ and the bound is trivial.

	The proof is by induction on $m$. The base case $m=1$ is Lemma~\ref{lem:main-induction-base}.

	For the step from $m$ to $m+1$, let $(\psi,A,B,L)$ be an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m+1,q,d)$ test, and let $k \ge 400(m+1)d$. For each $x \in \F_q$, let $(\psi,A^x,B^x,L^x)$ be the $x$-restricted strategy, with failure parameters $\eps_x,\delta_x,\gamma_x$. Apply the induction hypothesis to each restricted strategy with the same integer $k$ and define
	\[
		\nu_x = 1000k^2m^2\left(\eps_x^{1/1024}+\delta_x^{1/1024}+\gamma_x^{1/1024}+(d/q)^{1/1024}\right),
		\qquad
		\sigma_x = m^2\left(\nu_x + e^{-k/(80000m^2)}\right).
	\]
	This yields a measurement $G^x \in \polymeas{m}{q}{d}$ such that
	\[
		(A^x)_a^u \ot I \simeq_{\sigma_x} I \ot G_{[g(u)=a]}^x.
	\]

	Next define
	\[
		\zeta_x = 3000m\left(\eps_x^{1/32}+\delta_x^{1/32}+(d/q)^{1/32}\right).
	\]
	Theorem~\ref{thm:self-improvement-in-induction-section} produces a projective submeasurement $\widehat G^x \in \polysub{m}{q}{d}$ such that, for each $x$:
	\begin{enumerate}
		\item if $\widehat G^x=\sum_g \widehat G_g^x$, then
		      \[
			      \bra{\psi} \widehat G^x \ot I \ket{\psi} \ge (1-\sigma_x)-\zeta_x;
		      \]
		\item on average over $u \sim \F_q^m$,
		      \[
			      (A^x)_a^u \ot I \simeq_{\zeta_x} I \ot \widehat G_{[g(u)=a]}^x;
		      \]
		\item
		      \[
			      \widehat G_g^x \ot I \approx_{\zeta_x} I \ot \widehat G_g^x;
		      \]
		\item there exists a positive semidefinite operator $Z^x$ such that
		      \[
			      \bra{\psi} Z^x \ot (I-\widehat G^x) \ket{\psi} \le \zeta_x
		      \]
		      and for each $g \in \polyfunc{m}{q}{d}$,
		      \[
			      Z^x \ge \E_u (A^x)_{g(u)}^u.
		      \]
	\end{enumerate}

	We now average the slice-dependent parameters. By concavity of $\alpha \mapsto \alpha^c$ for $c \le 1$ and Lemma~\ref{lem:restricted-probabilities},
	\begin{align*}
		\E_x \nu_x
		 & \le 1000k^2m^2\left((\E_x \eps_x)^{1/1024}+(\E_x \delta_x)^{1/1024}+(\E_x \gamma_x)^{1/1024}+(d/q)^{1/1024}\right)                          \\
		 & \le 1000k^2m^2\left(\left(\frac{m+1}{m}\eps\right)^{1/1024}+\delta^{1/1024}+\left(\frac{m+1}{m}\gamma\right)^{1/1024}+(d/q)^{1/1024}\right) \\
		 & \le 1000k^2(m+1)^2\left(\eps^{1/1024}+\delta^{1/1024}+\gamma^{1/1024}+(d/q)^{1/1024}\right).
	\end{align*}
	Call this final quantity $\nu$. Then, defining
	\[
		\sigma = m^2\left(\nu + e^{-k/(80000m^2)}\right),
	\]
	we have
	\[
		\sigma \ge \E_x \sigma_x.
	\]
	Similarly,
	\begin{align*}
		\E_x \zeta_x
		 & \le 3000m\left((\E_x \eps_x)^{1/32}+(\E_x \delta_x)^{1/32}+(d/q)^{1/32}\right)         \\
		 & \le 3000m\left(\left(\frac{m+1}{m}\eps\right)^{1/32}+\delta^{1/32}+(d/q)^{1/32}\right) \\
		 & \le 3000(m+1)\left(\eps^{1/32}+\delta^{1/32}+(d/q)^{1/32}\right).
	\end{align*}
	Call this last quantity $\zeta$. For later use,
	\begin{align}
		\zeta
		 & = 3000(m+1)\left(\eps^{1/32}+\delta^{1/32}+(d/q)^{1/32}\right) \notag                              \\
		 & \le 1000k^2(m+1)^2\left(\eps^{1/32}+\delta^{1/32}+(d/q)^{1/32}\right) \notag                       \\
		 & \le 1000k^2(m+1)^2\left(\eps^{1/1024}+\delta^{1/1024}+\gamma^{1/1024}+(d/q)^{1/1024}\right) \notag \\
		 & = \nu. \label{eq:zeta-smaller-than-nu}
	\end{align}

	The averaged slice family $\{\widehat G^x\}_{x \in \F_q}$ therefore satisfies the four hypotheses of Theorem~\ref{thm:ld-pasting-in-induction-section} with completeness parameter $\kappa=\sigma+\zeta$. To verify the new $\nu$ bound in that theorem, observe that $(3000)^{1/32} \le 2$, so
	\[
		\zeta^{1/32} \le 2(m+1)\left(\eps^{1/1024}+\delta^{1/1024}+(d/q)^{1/1024}\right),
	\]
	and hence
	\begin{align*}
		 & 100k^2m\left(\eps^{1/32}+\delta^{1/32}+\gamma^{1/32}+\zeta^{1/32}+(d/q)^{1/32}\right)       \\
		 & \le 1000k^2(m+1)^2\left(\eps^{1/1024}+\delta^{1/1024}+\gamma^{1/1024}+(d/q)^{1/1024}\right) \\
		 & = \nu.
	\end{align*}
	Therefore Theorem~\ref{thm:ld-pasting-in-induction-section} gives a pasted measurement $H \in \polymeas{m+1}{q}{d}$ such that, on average over $u \sim \F_q^{m+1}$,
	\[
		A_a^u \ot I \simeq_{\sigma^*} I \ot H_{[h(u)=a]},
	\]
	where
	\[
		\sigma^* = (\sigma+\zeta)\left(1+\frac{1}{100m}\right) + 2\nu + e^{-k/(80000m^2)}.
	\]
	It remains to compare $\sigma^*$ with the next-stage target. The pasting
	parameter in Theorem~\ref{thm:ld-pasting-in-induction-section} is bounded by
	$\nu/5$ in this regime. Using this sharper estimate and
	\eqref{eq:zeta-smaller-than-nu}, we get
	\begin{align*}
		\sigma^*
		 & \le \left(m^2\left(\nu+e^{-k/(80000m^2)}\right)+\nu\right)
		      \left(1+\frac{1}{100m}\right) + \frac{2}{5}\nu
		      + e^{-k/(80000m^2)} \\
		 & = \left((m^2+1)\left(1+\frac{1}{100m}\right)+\frac{2}{5}\right)\nu
		     + \left(m^2\left(1+\frac{1}{100m}\right)+1\right)e^{-k/(80000m^2)}.
	\end{align*}
	For every $m\ge 1$,
	\[
		(m^2+1)\left(1+\frac{1}{100m}\right)+\frac{2}{5}\le (m+1)^2,
		\qquad
		m^2\left(1+\frac{1}{100m}\right)+1\le (m+1)^2.
	\]
	Therefore
	\[
		\sigma^* \le (m+1)^2\left(\nu+e^{-k/(80000m^2)}\right)
		\le (m+1)^2\left(\nu+e^{-k/(80000(m+1)^2)}\right),
	\]
	which is exactly the bound required for dimension $m+1$.
\end{proof}

\begin{proof}
	\proves{thm:main-formal}
	\uses{thm:main-induction, lem:symmetrization-bridge, lem:orthonormalization-main-lemma, prop:completing-to-measurement, prop:simeq-triangle-inequality, prop:simeq-data-processing, prop:simeq-to-approx, prop:triangle-sub, prop:triangle-inequality-for-approx_delta, lem:schwartz-zippel-individual, def:main-formal-envelope, def:main-formal-error-cascade, thm:sigma-bound-main-formal, thm:zeta-bounds-main-formal}
	Let
	\[
		(\psi,A^{\mathrm A},B^{\mathrm A},L^{\mathrm A},A^{\mathrm B},B^{\mathrm B},L^{\mathrm B})
	\]
	be a projective strategy passing the $(m,q,d)$ low individual degree test with probability at least $1-\eps$. Since the three subtests occur with probability $1/3$, this strategy is $(3\eps,3\eps,3\eps)$-good.

	Introduce a two-dimensional role register on each side and define the symmetrized state
	\[
		\ket{\psi_{\mathrm{sym}}}
		=
		\frac{1}{\sqrt{2}}
		\left(
		\ket{0}_{\mathrm{A}'}\ket{1}_{\mathrm{B}'}\ket{\psi}_{\mathrm{AB}}
		+
		\ket{1}_{\mathrm{A}'}\ket{0}_{\mathrm{B}'}\ket{\psi_{\mathrm{swap}}}_{\mathrm{AB}}
		\right).
	\]
	Define the symmetrized point measurement by
	\[
		(A_{\mathrm{sym}})_a^u
		=
		\ket{0}\bra{0} \ot A_a^{\mathrm A,u}
		+
		\ket{1}\bra{1} \ot A_a^{\mathrm B,u},
	\]
	and similarly for $B_{\mathrm{sym}}$ and $L_{\mathrm{sym}}$. This yields a symmetric $(3\eps,3\eps,3\eps)$-good strategy. Set
	\[
		\nu = 10000k^2m^2\left(\eps^{1/1024}+(d/q)^{1/1024}\right),
		\qquad
		\sigma = m^2\left(\nu+e^{-k/(80000m^2)}\right).
	\]
	Theorem~\ref{thm:main-induction} applied to the symmetrized strategy produces a measurement $G=\{G_g\} \in \polymeas{m}{q}{d}$ such that
	\begin{equation}\label{eq:just-applied-induction}
		(A_{\mathrm{sym}})_a^u \ot I \simeq_\sigma I \ot G_{[g(u)=a]},
		\qquad
		G_{[g(u)=a]} \ot I \simeq_\sigma I \ot (A_{\mathrm{sym}})_a^u.
	\end{equation}

	Now unsymmetrize by defining
	\[
		G_g^{\mathrm A} = (\bra{0} \ot I)\, G_g\, (\ket{0} \ot I),
		\qquad
		G_g^{\mathrm B} = (\bra{1} \ot I)\, G_g\, (\ket{1} \ot I).
	\]
	These are POVMs on the original prover space. Since $A_{\mathrm{sym}}$ is a convex combination of $(\ket{0}\bra{0} \ot I)$ and $(\ket{1}\bra{1} \ot I)$ weighted by the test probabilities, the cross-terms $\bra{0} \cdot A_{\mathrm{sym}} \cdot \ket{1}$ vanish. Expanding the consistency of $G$ gives
	\begin{gather}
		G_{[g(u)=a]}^{\mathrm A} \ot I \simeq_{2\sigma} I \ot A_a^{\mathrm B,u}, \label{eq:cons-a}\\
		I \ot G_{[g(u)=a]}^{\mathrm B} \simeq_{2\sigma} A_a^{\mathrm A,u} \ot I. \label{eq:cons-b}
	\end{gather}
	Since the original strategy is $(3\eps,3\eps,3\eps)$-good,
	\[
		A_a^{\mathrm A,u} \ot I \simeq_{3\eps} I \ot A_a^{\mathrm B,u}.
	\]
	Therefore Lemma~\ref{prop:simeq-triangle-inequality} yields
	\[
		G_{[g(u)=a]}^{\mathrm A} \ot I \simeq_{2\sigma + 2\sqrt{3\eps+2\sigma}} I \ot G_{[g(u)=a]}^{\mathrm B}.
	\]
	Expanding this consistency relation and applying Lemma~\ref{lem:schwartz-zippel-individual} to replace point evaluations by polynomial evaluations (at cost $md/q$) gives
	\begin{equation}\label{eq:G-self-consistency}
		G_g^{\mathrm A} \ot I \simeq_{\zeta_1} I \ot G_g^{\mathrm B},
	\end{equation}
