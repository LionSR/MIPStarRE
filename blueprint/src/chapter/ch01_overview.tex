\begin{theorem}[Raz--Safra]\label{thm:raz-safra}
	\lean{MIPStarRE.LDT.Test.razSafra}
	Suppose Provers~$\mathrm{A}$ and~$\mathrm{B}$ pass the $k=2$ surface-versus-point low-degree test with probability $1-\eps$. Then there exists a degree-$d$ polynomial $g \colon \F_q^m \to \F_q$ such that
	\[
		\Prb_{u \sim \F_q^m}[g(u)=a] \ge 1-\eps-\poly(m)\cdot\poly(d/q).
	\]
	Here $a$ denotes the prover's point answer.
\end{theorem}
\begin{remark}
	The current Lean declaration \lean{MIPStarRE.LDT.Test.razSafra} is a wrapper around the quoted classical Raz--Safra theorem: it takes the specialized auxiliary hypothesis \lean{MIPStarRE.LDT.Test.RazSafraSoundnessStatement} at the ambient point-answer function, error parameter, and chosen slack bound, whose premise is the explicit pass predicate \lean{MIPStarRE.LDT.Test.SurfaceVsPointPassCondition}. The surface test itself is modeled using parameterized genuine $2$-dimensional surfaces, i.e. ordered frames with linearly independent directions, together with a hidden parameter specifying the queried point on the surface, so the surface prover does not see that point directly. Its conclusion currently returns the repository-wide Lean type \lean{MIPStarRE.LDT.Polynomial}, which encodes individual degree at most $d$ rather than a separate total-degree-at-most-$d$ object. Accordingly, this result is not marked as formalized.
\end{remark}
\begin{proof}
	See Raz--Safra [RS97].
\end{proof}

\begin{theorem}[Polishchuk--Spielman]\label{thm:classical-test-soundness}
	\lean{MIPStarRE.LDT.Test.classicalTestSoundness}
	\uses{def:low-individual-degree-test}
	Suppose Provers~$\mathrm{A}$ and~$\mathrm{B}$ pass the low individual degree test with probability $1-\eps$. Then there exists a polynomial $g \colon \F_q^m \to \F_q$ with individual degree $d$ such that
	\[
		\Prb_{u \sim \F_q^m}[g(u)=a] \ge 1-\poly(m)\cdot\bigl(\poly(\eps)+\poly(d/q)\bigr).
	\]
	Here $a$ denotes the prover's point answer.
\end{theorem}
\begin{remark}
	The current Lean declaration \lean{MIPStarRE.LDT.Test.classicalTestSoundness} keeps the external Polishchuk--Spielman dependency explicit through a caller-chosen slack-bound parameter together with the auxiliary hypothesis \lean{MIPStarRE.LDT.Test.PolishchukSpielmanClassicalSoundnessStatement}. The repository no longer fixes any convenience placeholder formula for the schematic term \(\poly(m)\cdot(\poly(\eps)+\poly(d/q))\) appearing in the overview statement above: any concrete rate must instead be supplied by the specialized external hypothesis. Accordingly, this result is not marked as formalized.
\end{remark}
\begin{proof}
	See Polishchuk--Spielman [PS94].
\end{proof}

\begin{remark}[A degree-$(d+1)$ strategy that passes with probability $1-\frac{1}{m}$]\label{ex:bad-individual-degree-example}
		Fix the polynomial $h(x_1, \ldots, x_m) = x_1^{d+1}$. Consider the following classical strategy for the degree-$d$ low individual degree test. On a point query $u \in \F_q^m$, Prover~$\mathrm{A}$ answers $a = h(u)$. On an axis-parallel line query $\ell = \{u + x e_i : x \in \F_q\}$, Prover~$\mathrm{B}$ returns the restriction $h\vert_{\ell}$ when $i \neq 1$, since $h$ is then constant along $\ell$, and returns the zero polynomial when $i = 1$.

		The verifier certainly accepts whenever $i \neq 1$, which happens with probability $1 - \frac{1}{m}$. Thus the strategy passes the degree-$d$ test with probability at least $1-\eps$ for $\eps = \frac{1}{m}$. However, Prover~$\mathrm{A}$ is answering according to a polynomial whose individual degree in the first variable is $d+1$. By Schwartz--Zippel, any polynomial $g$ of individual degree at most $d$ agrees with $h$ on at most a $\frac{d+1}{q}$ fraction of all inputs, so
		\[
			\Prb_{u \sim \F_q^m}[g(u)=a] \le 1 - m\eps + \frac{d+1}{q}.
		\]
		This shows that the $m$-dependent slack in Theorems~\ref{thm:classical-test-soundness} and~\ref{thm:main-informal} is unavoidable up to polynomial factors.
\end{remark}

\begin{theorem}[Main theorem, informal]\label{thm:main-informal}
	\uses{def:low-individual-degree-test, def:projective-strategy, def:polymeasurements}
	If $d > 0$ and a two-prover projective strategy passes the $(m,q,d)$ low individual degree test with probability $1-\eps$, then there are global polynomial measurements whose evaluations agree with the point answers except with error $\poly(m)(\poly(\eps)+\poly(d/q))$.
\end{theorem}
