\chapter{Self-improvement}
\label{chap:self-improvement}

Throughout this chapter, $(\psi,A,B,L)$ is an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m,q,d)$ low individual degree test.


\begin{lemma}[Self-improvement with non-projective output]\label{lem:self-improvement-helper}
  \lean{MIPStarRE.Paper2009LDT.Section9SelfImprovement.selfImprovementHelper}
  \uses{def:polymeasurements, def:simeq, def:strong-self-consistency}
  Let $G \in \polymeas{m}{q}{d}$ be a measurement such that, on average over $u \sim \F_q^m$,
  \[
    A_a^u \ot I \simeq_\nu I \ot G_{[g(u)=a]}.
  \]
  Set
  \[
    \zeta = 100m\left(\eps^{1/2}+\delta^{1/2}+(d/q)^{1/2}\right).
  \]
  Then there exists a submeasurement $H \in \polysub{m}{q}{d}$ such that:
  \begin{enumerate}
    \item if $H=\sum_h H_h$, then $\bra{\psi} H \ot I \ket{\psi} \ge (1-\nu)-\zeta$;
    \item on average over $u \sim \F_q^m$,
    \[
      A_a^u \ot I \simeq_\zeta I \ot H_{[h(u)=a]};
    \]
    \item the family $H$ is $\zeta$-strongly self-consistent;
    \item there exists a positive semidefinite operator $Z$ such that
    \[
      \bra{\psi} Z \ot I \ket{\psi} - \E_u \sum_a \bra{\psi} A_a^u \ot H_{[h(u)=a]} \ket{\psi} \le \zeta,
    \]
    and $Z \ge \E_u A_{h(u)}^u$ for every $h \in \polyfunc{m}{q}{d}$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  \uses{lem:sdp, lem:add-in-u, lem:global-variance-of-points, prop:simeq-to-approx}
  Solve the SDP from Lemma~\ref{lem:sdp} to obtain an optimal measurement $T=\{T_h\}$ and define
  \[
    H_h = \E_u A^u_{h(u)} T_h A^u_{h(u)}.
  \]
  The four conclusions are proved separately. For completeness, apply Lemma~\ref{lem:add-in-u} with $M=A$ and the diagonal set $\{(a,h):h(u)=a\}$, then use the dual lower bound $Z\ge A_h$ together with complementary slackness. For consistency with $A$, apply Lemma~\ref{lem:add-in-u} with the off-diagonal set $\{(a,h):h(u)\ne a\}$; the right-hand side vanishes because the point measurements $A^u$ are projective. For strong self-consistency, apply Lemma~\ref{lem:add-in-u} with $M=H$ and then bound the off-diagonal polynomial pairs by global variance and Schwartz--Zippel. For boundedness, use the same averaged expression as in the completeness step but keep the dual witness $Z$ on the first tensor factor. Thus the source really branches into four quantitative estimates rather than a single uniform argument.
\end{proof}


\begin{lemma}[Dual SDP and complementary slackness]\label{lem:sdp}
  \lean{MIPStarRE.Paper2009LDT.Section9SelfImprovement.sdp}
  Let
  \[
    A_g = \E_{u \sim \F_q^m} A^u_{g(u)}.
  \]
  Consider the primal program
  \[
    \sup \sum_g \Tr(T_g A_g)
    \qquad\text{subject to } T_g \ge 0 \text{ for all } g, \sum_g T_g \le I,
  \]
  and the dual program
  \[
    \inf \Tr(Z)
    \qquad\text{subject to } Z \ge A_g \text{ for all } g.
  \]
  These programs are dual to each other. Moreover there are optimal solutions with $\sum_g T_g = I$ and
  \[
    T_g Z = T_g A_g
  \]
  for every $g$.
\end{lemma}
\begin{proof}
  Rewrite the primal in block-matrix canonical form so that one block corresponds to each polynomial $g$. Slater's condition is immediate from a strictly feasible primal point and a sufficiently large dual matrix, so strong duality holds. Complementary slackness for the optimal pair then forces the slack block attached to $I-\sum_g T_g$ to vanish, giving $\sum_g T_g=I$, and also forces each block $T_g(Z-A_g)$ to be zero, which is the stated identity.
\end{proof}

\begin{lemma}[Averaging over the evaluation point]\label{lem:add-in-u}
  \lean{MIPStarRE.Paper2009LDT.Section9SelfImprovement.addInU}
  \uses{lem:global-variance-of-points, prop:simeq-to-approx}
  Let $T=\{T_h\}$ be an optimal solution of the primal SDP, let
  \[
    H_h = \E_{u \sim \F_q^m} A^u_{h(u)} T_h A^u_{h(u)},
  \]
  and set
  \[
    \zeta_{\mathrm{var}} = 24m\left(\eps+\delta+\frac{md}{q}\right).
  \]
  If $M=\{M_o^u\}$ is any submeasurement and $S_u \subseteq \mathcal O \times \polyfunc{m}{q}{d}$ is any family of subsets, then
  \[
    \E_u \sum_{(o,h) \in S_u} \bra{\psi} M_o^u \ot H_h \ket{\psi}
    \approx_{4\sqrt{\zeta_{\mathrm{var}}}}
    \E_u \sum_{(o,h) \in S_u} \bra{\psi} A^u_{h(u)} M_o^u A^u_{h(u)} \ot T_h \ket{\psi}.
  \]
\end{lemma}
\begin{proof}
  Expand $H_h$ by introducing a second point $v$. Two Cauchy--Schwarz steps move the two factors $A^v_{h(v)}$ from Bob's side to Alice's side, paying only the self-consistency error of $A$. Then use Lemma~\ref{lem:global-variance-of-points} twice to replace $A^v_{h(v)}$ by $A^u_{h(u)}$ while keeping $T_h$ fixed. The resulting error is $2\sqrt{2\delta}+2\sqrt{\zeta_{\mathrm{var}}}$, which the source simplifies to $4\sqrt{\zeta_{\mathrm{var}}}$ because $2\delta\le \zeta_{\mathrm{var}}$.
\end{proof}

\begin{theorem}[Self-improvement]\label{thm:self-improvement}
  \lean{MIPStarRE.Paper2009LDT.Section9SelfImprovement.selfImprovement}
  \uses{def:polymeasurements, def:simeq, def:strong-self-consistency}
  Let $G \in \polymeas{m}{q}{d}$ be a measurement such that, on average over $u \sim \F_q^m$,
  \[
    A_a^u \ot I \simeq_\nu I \ot G_{[g(u)=a]}.
  \]
  Set
  \[
    \zeta = 3000m\left(\eps^{1/32}+\delta^{1/32}+(d/q)^{1/32}\right).
  \]
  Then there exists a projective submeasurement $H \in \polysub{m}{q}{d}$ such that:
  \begin{enumerate}
    \item if $H=\sum_h H_h$, then $\bra{\psi} H \ot I \ket{\psi} \ge (1-\nu)-\zeta$;
    \item on average over $u \sim \F_q^m$,
    \[
      A_a^u \ot I \simeq_\zeta I \ot H_{[h(u)=a]};
    \]
    \item $H_h \ot I \approx_\zeta I \ot H_h$;
    \item there exists a positive semidefinite operator $Z$ such that
    \[
      \bra{\psi} Z \ot (I-H) \ket{\psi} \le \zeta,
    \]
    and $Z \ge \E_u A_{h(u)}^u$ for every $h \in \polyfunc{m}{q}{d}$.
  \end{enumerate}
\end{theorem}
\begin{proof}
  \uses{lem:self-improvement-helper, thm:orthonormalization, prop:self-consistency-implies-data-processing, prop:completeness-transfer-self-consistent-A, prop:triangle-sub, prop:two-notions-of-self-consistency}
  Apply Lemma~\ref{lem:self-improvement-helper} to obtain a strongly self-consistent submeasurement $\widehat H$ with error
  \[
    \widehat\zeta = 100m\left(\eps^{1/2}+\delta^{1/2}+(d/q)^{1/2}\right).
  \]
  Orthogonalization gives a nearby projective submeasurement $H$ with error $\widehat\zeta_{\mathrm{ortho}}=100\widehat\zeta^{1/4}$, and Lemma~\ref{prop:self-consistency-implies-data-processing} upgrades this to a post-processed comparison error $\widehat\zeta_{\mathrm{dataprocess}}=8\widehat\zeta+8\sqrt{\widehat\zeta_{\mathrm{ortho}}}$. The four properties are then transferred one by one: completeness by Lemma~\ref{prop:completeness-transfer-self-consistent-A}, consistency by Lemma~\ref{prop:triangle-sub}, self-consistency by Lemma~\ref{prop:two-notions-of-self-consistency} plus the triangle inequality for $\approx_\delta$, and boundedness by replacing $\widehat H_{[h(u)=a]}$ with $H_{[h(u)=a]}$ in the helper bound. The source keeps the intermediate errors explicit and checks that each of
  \[
    2\widehat\zeta+2\sqrt{\widehat\zeta_{\mathrm{ortho}}}, \qquad
    \widehat\zeta+\sqrt{\widehat\zeta_{\mathrm{dataprocess}}}, \qquad
    6\widehat\zeta+6\widehat\zeta_{\mathrm{ortho}}
  \]
  is at most $3000m(\eps^{1/32}+\delta^{1/32}+(d/q)^{1/32})$.
\end{proof}
