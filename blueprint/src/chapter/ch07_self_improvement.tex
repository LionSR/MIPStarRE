\chapter{Self-improvement}
\label{chap:self-improvement}

Throughout this chapter, $(\psi,A,B,L)$ is an $(\eps,\delta,\gamma)$-good symmetric strategy for the $(m,q,d)$ low individual degree test.

\section{Non-projective Output}
\label{sec:self-improvement}

A large part of this chapter is devoted to proving
Lemma~\ref{lem:self-improvement-helper}, which is a slightly weaker form of the
projective self-improvement theorem proved in Section~\ref{sec:self-improvement-projective}.
The key difference is that the output family~$H$ is only required to be a
submeasurement, rather than a projective submeasurement. Once this non-projective
statement is available, Theorem~\ref{thm:orthonormalization} upgrades~$H$ to a
projective family without losing control of the four quantitative conclusions.

Compared with the projective theorem, the helper lemma records strong
self-consistency using the explicit inequality from
Definition~\ref{def:strong-self-consistency}, keeps the boundedness conclusion in
terms of an auxiliary operator~$Z$, and produces a much smaller error parameter.

\begin{lemma}[Self-improvement with non-projective output]\label{lem:self-improvement-helper}
  \lean{MIPStarRE.LDT.SelfImprovement.selfImprovementHelper}
  \leanok
  \uses{def:polymeasurements, def:simeq, def:strong-self-consistency}
  Let $G \in \polymeas{m}{q}{d}$ be a measurement with the following property:
  \begin{enumerate}
    \item (Consistency with~$A$): \label{item:self-improvement-G-consistency} On average over $u \sim \F_q^{m}$,
    \[
      A^{u}_a \ot I \simeq_{\nu} I \ot G_{[g(u)=a]}.
    \]
  \end{enumerate}
  Let
  \[
    \zeta = 100m\cdot \Big(\eps^{1/2} + \delta^{1/2} + (d/q)^{1/2}\Big).
  \]
  Then there exists $H \in \polysub{m}{q}{d}$ with the following properties:
  \begin{enumerate}
    \item (Completeness): \label{item:self-improvement-completeness} If $H = \sum_h H_h$, then
    \[
      \bra{\psi} H \ot I \ket{\psi} \geq (1-\nu)-\zeta.
    \]
    \item (Consistency with~$A$): \label{item:self-improvement-A-consistency} On average over $u \sim \F_q^m$,
    \[
      A^u_a \ot I \simeq_{\zeta} I \ot H_{[h(u) = a]}.
    \]
    \item (Strong self-consistency): \label{item:self-improvement-self}
    \[
      \sum_{h} \bra{\psi} H_h \ot H_h \ket{\psi} \geq \bra{\psi} H \ot I \ket{\psi} - \zeta.
    \]
    \item (Boundedness): \label{item:self-improvement-boundedness} There exists a positive-semidefinite matrix~$Z$ such that
    \[
      \bra{\psi} Z \ot I \ket{\psi} -\E_{u} \sum_a \bra{\psi}  A^{u}_{a} \ot H_{[h(u)=a]} \ket{\psi} \leq \zeta
    \]
    and for each $h \in \polyfunc{m}{q}{d}$,
    \[
      Z \geq \left(\E_{u} A^{u}_{h(u)}\right).
    \]
  \end{enumerate}
\end{lemma}
\begin{proof}
  \uses{lem:sdp, lem:add-in-u, lem:global-variance-of-points, prop:simeq-to-approx, prop:simeq-for-measurements}
  \leanok
  Let $T=\{T_g\}$ and $Z$ be the optimal solutions to the SDPs~\eqref{eq:primal-objective} and~\eqref{eq:dual-objective} given by Lemma~\ref{lem:sdp}. Then $T$ is a measurement, and
  \begin{align}
    \forall g \in \polyfunc{m}{q}{d}: \qquad
    Z &\geq \left(\E_{u} A^{u}_{g(u)}\right), \label{eq:Z-greater-than-A}\\
    T_g \cdot Z &= T_g \cdot \left(\E_{u} A^{u}_{g(u)}\right). \label{eq:swap-Z-for-A}
  \end{align}
  For each $u \in \F_q^m$, define $H^u=\{H_h^u\}_{h \in \polyfunc{m}{q}{d}}$ by
  \[
    H_h^u := A^u_{h(u)} \cdot T_h \cdot A^u_{h(u)},
  \]
  and let
  \[
    H_h := \E_{u} H_h^u.
  \]
  The pointwise family $H^u$ is a submeasurement because
  \[
    \sum_h H_h^u
    = \sum_a A_a^u \cdot \Big(\sum_{h:h(u)=a} T_h\Big)\cdot A_a^u
    \le \sum_a (A_a^u)^2
    = I,
  \]
  where the inequality uses that~$T$ is a measurement and the last identity uses
  that~$A$ is projective. Averaging over~$u$ gives
  \[
    \sum_h H_h
    = \E_{u} \sum_h H_h^u
    \le I,
  \]
  so $H$ is also a submeasurement and therefore lies in~$\polysub{m}{q}{d}$.

  Set
  \[
    \zeta_{\mathrm{variance}} = 24m\cdot\Big(\eps + \delta + \frac{md}{q}\Big).
  \]
  Lemma~\ref{lem:add-in-u} below is the technical transfer from the averaged family~$H$ back to the sandwiched operators $A^u_{h(u)}T_hA^u_{h(u)}$. We use it in the completeness, consistency, strong self-consistency, and boundedness estimates that follow.

  \emph{Proof of \ref{item:self-improvement-self}.}
  From the definition of $H_h^u$ and the projectivity of~$A$,
  \begin{align}
    H^u_h &= A^u_{h(u)} \cdot T_h \cdot A^u_{h(u)} = A^u_{h(u)} \cdot H^u_h \cdot A^u_{h(u)}, \label{eq:h-sandwich}\\
    A^u_{h(u)} \cdot H^u_{h'} \cdot A^u_{h(u)} &= H^u_{h'} \cdot A^u_{h(u)} = \bigl(A^u_{h(u)} \cdot T_{h'} \cdot A^u_{h(u)}\bigr)\cdot \bone{h(u) = h'(u)}. \label{eq:h-blt}
  \end{align}

    \sum_{h\in\polyfunc{m}{q}{d}} \bra{\psi} H_h \ot H_h \ket{\psi}

  \end{align}

    \approx_{\sqrt{\zeta_{\mathrm{variance}}}}
    \E_{u,v} \sum_{h \neq h'} \bra{\psi} \bigl(A^{v}_{h(v)} \cdot T_{h'} \cdot A^{u}_{h(u)}\bigr) \ot T_h \ket{\psi} \cdot \bone{h(u) = h'(u)}.

    \Big|\E_{u,v} \sum_{h \neq h'} \bra{\psi} \bigl((A^{u}_{h(u)} - A^{v}_{h(v)}) \cdot T_{h'} \cdot A^{u}_{h(u)}\bigr) \ot T_h \ket{\psi} \cdot \bone{h(u) = h'(u)}\Big|\\

  \emph{Proof of \ref{item:self-improvement-boundedness}.}
  The helper-stage boundedness quantity is
  \[
    \bra{\psi} Z \ot I \ket{\psi} -\E_{u} \sum_a \bra{\psi}  A^{u}_{a} \ot H_{[h(u)=a]} \ket{\psi}.
  \]
  The reindexing of the sum by~$h$ is the algebraic identity
  \lean{MIPStarRE.LDT.SelfImprovement.helperAgreementOperatorAtPoint_eq_sum_polynomial}
  \[
    \sum_a A^u_a \ot H_{[h(u)=a]} = \sum_h A^u_{h(u)} \ot H_h,
  \]
  whose averaged scalar form reads
  \lean{MIPStarRE.LDT.SelfImprovement.helper_agreement_average_ev_eq_polynomial_sum}
  \[
    \E_u \sum_a \bra{\psi}\,A^u_a \ot H_{[h(u)=a]}\,\ket{\psi}
      = \E_u \sum_h \bra{\psi}\,A^u_{h(u)} \ot H_h\,\ket{\psi}.
  \]
  Using $\sum_a A_a^u=I$, the pointwise difference between the right-placed
  total $I \ot H.\mathrm{total} = \sum_h I \ot H_h$ and the helper-agreement
  operator at $u$ reindexes as the off-diagonal sum
  \lean{MIPStarRE.LDT.SelfImprovement.helperAgreementOperatorAtPoint_off_diagonal_decomposition}
  \[
    \sum_h I \ot H_h - \sum_a A^u_a \ot H_{[h(u)=a]}
      = \sum_h \sum_{a \neq h(u)} A^u_a \ot H_h,
  \]
  whose averaged scalar form reads
  \lean{MIPStarRE.LDT.SelfImprovement.helper_boundedness_slack_average_ev_eq_off_diagonal_avg}
  \[
    \sum_h \bra{\psi}\,I \ot H_h\,\ket{\psi}
        - \E_u \sum_a \bra{\psi}\,A^u_a \ot H_{[h(u)=a]}\,\ket{\psi}
      = \E_u \sum_h \sum_{a \neq h(u)} \bra{\psi}\,A^u_a \ot H_h\,\ket{\psi}.
  \]
  Combined with~\eqref{eq:explicit-bound-for-A-consistency} this gives
  \lean{MIPStarRE.LDT.SelfImprovement.helper_boundedness_gap_eq_upper_gap_add_off_diagonal_avg}
  \lean{MIPStarRE.LDT.SelfImprovement.helper_boundedness_gap_le_selfImprovementHelperError}
  \lean{MIPStarRE.LDT.SelfImprovement.helper_boundedness_gap_le_selfImprovementHelperError_of_pointConsistencyAddInU_transfer}
  \lean{MIPStarRE.LDT.SelfImprovement.helper_upper_gap_rightTensor_le_three_sqrt_delta_of_helper_outputs}
  \lean{MIPStarRE.LDT.SelfImprovement.helper_boundedness_gap_le_selfImprovementHelperError_of_helper_outputs}
  \[
    \bra{\psi} Z \ot I \ket{\psi}
    -\E_{u} \sum_a \bra{\psi}  A^{u}_{a} \ot H_{[h(u)=a]} \ket{\psi}
    \le \bra{\psi} Z \ot I \ket{\psi} - \sum_h \bra{\psi} I \ot H_h \ket{\psi} + 4 \sqrt{\zeta_{\mathrm{variance}}}.
  \]
  By~\eqref{eq:h-versus-z-lower-bound}, the right-hand side is at most $3\sqrt{\delta}+4\sqrt{\zeta_{\mathrm{variance}}}$. Substituting the definition of $\zeta_{\mathrm{variance}}$ and collecting terms gives
  \lean{MIPStarRE.LDT.SelfImprovement.helper_boundedness_error_le_selfImprovementHelperError}
  \[
    23m\cdot\Big(\eps^{1/2} + \delta^{1/2} + (d/q)^{1/2}\Big)
    \le \zeta.
  \]
  Together with~\eqref{eq:Z-greater-than-A}, this proves \ref{item:self-improvement-boundedness}.
\end{proof}

  \end{align}

  \end{align}

  \end{equation}

\begin{proof}
  \leanok
  Rewrite~\eqref{eq:primal-objective} in canonical block form. Let $r$ be the dimension of the space on which~$A$ acts, let $M = |\polyfunc{m}{q}{d}|$, and fix an ordering $g_1,\ldots,g_M$ of the polynomials. Consider
  \begin{align}
    \sup &\quad \Tr(C^\dagger X) \label{eq:primal-canonical}\\
    \text{s.t.} &\quad \Tr(D_{ij}^\dagger X) = b_{ij} \qquad\forall i,j\in\{1,\ldots,r\}, \notag\\
    &\quad X \geq 0, \notag
  \end{align}
  where
  \[
    C = \sum_{i = 1}^M \ket{i}\bra{i} \ot A_{g_i},
  \]
  and, for $i,j \in \{1,\ldots,r\}$,
  \[
    D_{ij} = \sum_{k=1}^{M+1} \ket{k}\bra{k} \ot \ket{i}\bra{j},
    \qquad
    b_{ij}  = \left\{\begin{array}{rl}
      1 & \text{if } i = j,\\
      0 & \text{otherwise}.
    \end{array}\right.
  \]
  Writing $X=\sum_{i,j=1}^{M+1}\ket{i}\bra{j}\ot X_{ij}$, the constraints say precisely that $\sum_{i=1}^{M+1} X_{ii}=I$, hence $\sum_{i=1}^{M}X_{ii}\le I$, and the objective is
  \[
    \Tr(C^\dagger X)=\sum_{i=1}^M \Tr(A_{g_i}\cdot X_{ii}).
  \]
  Thus~\eqref{eq:primal-canonical} is equivalent to~\eqref{eq:primal-objective} under the identification $T_{g_i}=X_{ii}$.

  The canonical dual is
  \begin{align}
    \inf &\quad \sum_i\, z_{ij} b_{ij} \label{eq:dual-canonical}\\
    \text{s.t.} &\quad  \sum_{i,j} \,z_{ij} D_{ij} \geq C. \label{eq:dual-canonical-constraint}
  \end{align}
  Since
  \[
    \sum_{i,j=1}^r z_{ij} D_{ij}
    = \sum_{k=1}^{M+1} \ket{k}\bra{k} \ot \left(\sum_{i,j=1}^r z_{ij}\ket{i}\bra{j}\right)
    =: \sum_{k=1}^{M+1}\ket{k}\bra{k} \ot Z,
  \]
  the constraint~\eqref{eq:dual-canonical-constraint} is exactly $Z\ge A_{g_i}$ for every $i$, and the objective is $\Tr(Z)$.  In Lean, the latter statement is recorded as the trace-pairing identity
  \[
    \Re\Tr(D(Z)X)=\Re\Tr(Z)
  \]
  for every feasible canonical primal matrix \(X\), where \(D(Z)\) denotes the
  block-diagonal canonical dual operator.  Consequently the canonical
  primal-dual gap is the real trace pairing of \(X\) with the canonical dual
  slack \(D(Z)-C\).  Since the real trace pairing of positive semidefinite
  operators is nonnegative, Lean also records the resulting weak-duality
  inequality for any feasible canonical primal and dual pair.
  Hence~\eqref{eq:dual-canonical} is equivalent to~\eqref{eq:dual-objective}.

  Slater's condition holds because
  \[
    T_g = \frac{1}{2M}\cdot I \qquad\forall g\in\polyfunc{m}{q}{d},
    \qquad
    Z = 2I
  \]
  are strictly feasible for the primal and dual respectively. Therefore strong duality holds. For an optimal pair $(X,(z_{ij}))$ in canonical form, complementary slackness gives
  \begin{equation}\label{eq:complementary-slackness}
    X\Big( \sum_{i,j} \,z_{ij} D_{ij} - C \Big) \,=\, 0.
  \end{equation}
  We may assume that $X$ is block diagonal.
  Translating~\eqref{eq:complementary-slackness} back to the variables
  $\{T_g\}$ and~$Z$ yields $T_g(Z-A_g)=0$ for every~$g$ and \(S Z=0\), where
  \(S=I-\sum_gT_g\) is the slack block.
  The paper then treats this as forcing \(S=0\).  The Lean development now
  records this saturation as part of the canonical optimal-pair datum for the
  block SDP.  In the concrete matrix realization, Lean records the defect-zero
  condition as
  \texttt{matrixSdpComplementarySlacknessDefect}; the theorem
  \texttt{MatrixSdpOptimalWitness.complementarySlacknessEquation} translates it
  to the displayed equation \(T_gZ=T_gA_g\), while
  \texttt{MatrixSdpOptimalWitness.primalMeasurement} expresses
  \(\sum_g T_g=I\) as a complete measurement.  The statement
  \texttt{MatrixSdpStatementWithSlackness} records this matrix-level
  strong-duality output and the abstract theorem
  \texttt{sdp\_statement\_with\_slackness} specializes it to the paper's
  self-improvement setting.  The explicit Slater witnesses used above are
  separately recorded by \texttt{matrixSdpStrictPrimalSubmeasurement},
  \texttt{matrixSdpStrictDualWitness}, and
  \texttt{matrixSdpFeasibleBounds\_canonical}; the dual feasibility of \(2I\)
  follows from \texttt{matrixAveragedPointOperator\_le\_one}.
  The block-diagonal reduction is also recorded for arbitrary feasible
  canonical matrices: extracting the diagonal blocks preserves the objective
  and transfers canonical complementary slackness to the extracted paper primal
  submeasurement.
\end{proof}

  \begin{align*}
    &\E_{u,v} \sum_{a \in \F_q} \bra{\psi} \bigl(A^v_a \ot I - I \ot A^v_a\bigr)
      \cdot \Bigl(\sum_{(o,h)\in S_u,\, h(v)=a} M^u_o \ot T_h\Bigr)
      \cdot \bigl(A^v_a \ot I - I \ot A^v_a\bigr)\ket{\psi} \\
    &\le \E_{u,v} \sum_{a \in \F_q} \bra{\psi} \bigl(A^v_a \ot I - I \ot A^v_a\bigr)^2
      \ket{\psi},
  \end{align*}
  where the inequality uses that $M^u$ and $T$ are submeasurements. By
  Lemma~\ref{prop:simeq-to-approx} and the self-consistency of~$A$, this is at most
  $2\delta$. The second factor is
  \[
    \E_{u,v} \sum_{(o,h)\in S_u} \bra{\psi} M^u_o \ot H^v_h \ket{\psi},
  \]
  which is at most~$1$ because $M^u$ and $H^v$ are submeasurements.

  Next, we claim that
  \begin{equation}\label{eq:move-second-point-projector}
    \eqref{eq:move-first-point-projector}
    \approx_{\sqrt{2\delta}}
    \E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl(A^{v}_{h(v)} \cdot M^{u}_o \cdot A^{v}_{h(v)}\bigr) \ot T_h  \ket{\psi}.
  \end{equation}
  Again, we bound the magnitude of the difference:
  \begin{align}\label{eq:move-second-point-projector-cauchy-schwarz}
    &\Big|\E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl((A^{v}_{h(v)} \cdot M^{u}_o) \ot T_h\bigr) \cdot \bigl(A^{v}_{h(v)} \ot I - I \ot A^{v}_{h(v)}\bigr)  \ket{\psi}\Big| \notag\\
    &\leq \sqrt{\E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl(A^{v}_{h(v)} \cdot M^{u}_o \cdot A^{v}_{h(v)}\bigr) \ot T_h  \ket{\psi}} \\
    &\qquad\cdot \sqrt{\E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl(A^{v}_{h(v)} \ot I - I \ot A^{v}_{h(v)}\bigr) \cdot (M^{u}_o \ot T_h) \cdot \bigl(A^{v}_{h(v)} \ot I - I \ot A^{v}_{h(v)}\bigr) \ket{\psi}}. \notag
  \end{align}
  The first factor is bounded by
  \begin{align*}
    &\E_{u,v} \sum_h \bra{\psi}
      \Bigl(A^v_{h(v)} \cdot \sum_{o:(o,h)\in S_u} M^u_o \cdot A^v_{h(v)}\Bigr)
      \ot T_h \ket{\psi} \\
    &\le \E_{u,v} \sum_h \bra{\psi} (A^v_{h(v)})^2 \ot T_h \ket{\psi}
      \le \E_{u,v} \sum_h \bra{\psi} I \ot T_h \ket{\psi}
      \le 1,
  \end{align*}
  using first that $M$ is a submeasurement, then that $A^v_{h(v)} \le I$, and
  finally that $T$ is a submeasurement ($\sum_g T_g \le I$). The second factor is
  exactly the same expression
  as in~\eqref{eq:move-first-point-projector-cauchy-schwarz}, hence at most
  $\sqrt{2\delta}$.

  Having moved both copies of~$A$ to the left-hand side, we next show that
  \begin{equation}\label{eq:replace-left-point-conditioner}
    \eqref{eq:move-second-point-projector}
    \approx_{\sqrt{\zeta_{\mathrm{variance}}}}
    \E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl(A^{u}_{h(u)} \cdot M^{u}_o \cdot A^{v}_{h(v)}\bigr) \ot T_h  \ket{\psi}.
  \end{equation}
  The difference is bounded by
  \begin{multline}\label{eq:replace-left-point-conditioner-cauchy-schwarz}
    \Big|\E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl((A^{v}_{h(v)} - A^{u}_{h(u)}) \cdot M^{u}_o \cdot A^{v}_{h(v)}\bigr) \ot T_h  \ket{\psi}\Big|\\
    \leq
    \sqrt{\E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl((A^{v}_{h(v)} - A^{u}_{h(u)}) \cdot M^{u}_o \cdot (A^{v}_{h(v)} - A^{u}_{h(u)})\bigr) \ot T_h  \ket{\psi}}\\
    \cdot \sqrt{\E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl(A^{v}_{h(v)} \cdot M^{u}_o \cdot A^{v}_{h(v)}\bigr) \ot T_h  \ket{\psi}}.
  \end{multline}
  The first factor can be rewritten as
  \begin{align*}
    &\E_{u,v} \sum_h \bra{\psi}
      \Bigl((A^v_{h(v)} - A^u_{h(u)}) \cdot
      \sum_{o:(o,h)\in S_u} M^u_o \cdot
      (A^v_{h(v)} - A^u_{h(u)})\Bigr) \ot T_h \ket{\psi} \\
    &\le \E_{u,v} \sum_h \bra{\psi}
      (A^v_{h(v)} - A^u_{h(u)})^2 \ot T_h \ket{\psi},
  \end{align*}
  because $M^u$ is a submeasurement. Lemma~\ref{lem:global-variance-of-points}
  bounds this by~$\zeta_{\mathrm{variance}}$ since $T \in \polysub{m}{q}{d}$. The
  second factor is exactly the first factor
  from~\eqref{eq:move-second-point-projector-cauchy-schwarz},
  so it is at most~$1$.

  Finally, we show that
  \begin{equation}\label{eq:replace-right-point-conditioner}
    \eqref{eq:replace-left-point-conditioner}
    \approx_{\sqrt{\zeta_{\mathrm{variance}}}}
    \E_{u,v} \sum_{(o, h) \in S_{u}} \bra{\psi} \bigl(A^{u}_{h(u)} \cdot M^{u}_o \cdot A^{u}_{h(u)}\bigr) \ot T_h  \ket{\psi}.
  \end{equation}
  The same Cauchy--Schwarz estimate
  as in~\eqref{eq:replace-left-point-conditioner-cauchy-schwarz}, with
  the final copy of $A^v_{h(v)}$ replaced by $A^u_{h(u)}$, gives the same bound
  $\sqrt{\zeta_{\mathrm{variance}}}$. Indeed, the factor containing
  $A^u_{h(u)} \cdot M^u_o \cdot A^u_{h(u)}$ is bounded by~$1$ by the same
  submeasurement argument as above, while the factor containing
  $(A^v_{h(v)} - A^u_{h(u)})^2$ is unchanged. Summing the four transports gives an
  error of $2\sqrt{2\delta} + 2\sqrt{\zeta_{\mathrm{variance}}}$, and the lemma
  follows from $2\delta \leq \zeta_{\mathrm{variance}}$.
\end{proof}

\begin{theorem}[Self-improvement]\label{thm:self-improvement}
  \lean{MIPStarRE.LDT.SelfImprovement.selfImprovement}
  % No \leanok is attached here: the current Lean proof still depends
  % on the residual-domination obligation tracked by issue #1642.
  \uses{def:polymeasurements, def:simeq, def:approx_delta, def:strong-self-consistency}
  Let $G \in \polymeas{m}{q}{d}$ be a measurement with the following properties:
  \begin{enumerate}
    \item (Consistency with~$A$): \label{item:self-improvement-projective-G-consistency} On average over $u \sim \F_q^{m}$,
    \[
      A^{u}_a \ot I \simeq_{\nu} I \ot G_{[g(u)=a]}.
    \]
  \end{enumerate}
  Let
  \[
    \zeta = 3000m\cdot \Big(\eps^{1/32} + \delta^{1/32} + (d/q)^{1/32}\Big).
  \]
  Then there exists a projective submeasurement $H \in \polysub{m}{q}{d}$ with the following properties:
  \begin{enumerate}
    \item (Completeness): \label{item:self-improvement-projective-completeness} If $H = \sum_h H_h$, then
    \[
      \bra{\psi} H \ot I \ket{\psi} \geq (1-\nu)-\zeta.
    \]
    \item (Consistency with~$A$): \label{item:self-improvement-projective-A-consistency} On average over $u \sim \F_q^m$,
    \[
      A^u_a \ot I \simeq_{\zeta} I \ot H_{[h(u) = a]}.
    \]
    \item (Strong self-consistency): \label{item:self-improvement-projective-self}
    \[
      H_h \ot I \approx_{\zeta} I \ot H_h.
    \]
    \item (Boundedness): \label{item:self-improvement-projective-boundedness} There exists a positive-semidefinite matrix~$Z$ such that
    \[
      \bra{\psi} Z \ot (I - H) \ket{\psi} \leq \zeta
    \]
    and for each $h \in \polyfunc{m}{q}{d}$,
    \[
      Z \geq \left(\E_{u} A^{u}_{h(u)}\right).
    \]
  \end{enumerate}
\end{theorem}
